\begin{tcolorbox}[colback=sky, colframe=navy, boxrule=0.9pt, arc=2mm,
    left=4mm, right=4mm, top=3mm, bottom=3mm]
\small A school nurse recorded the number of hours of sleep reported by
\textbf{8 students} the night before a big exam.
The sorted data are:
\[4,\quad 7,\quad 7,\quad 9,\quad 10,\quad 13,\quad 15,\quad 30\]
\end{tcolorbox}

\vspace{0.18in}

\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item Find the \textbf{mean}. Show your calculation.\\[4pt]
        \ans{$\bar{x} = \dfrac{4+7+7+9+10+13+15+30}{8} = \dfrac{95}{8} = \mathbf{11.875}$ hours}

  \item Find the \textbf{median}. Show your work.\\[4pt]
        \ans{$n = 8$ (even); middle values are the 4th and 5th: 9 and 10.\\
        Median $= \dfrac{9+10}{2} = \mathbf{9.5}$ hours.}

  \item Find $Q_1$, $Q_3$, and the \textbf{IQR}. Show your work.\\[4pt]
        \ans{Lower half: 4, 7, 7, 9 $\Rightarrow$ $Q_1 = \dfrac{7+7}{2} = 7$\\
        Upper half: 10, 13, 15, 30 $\Rightarrow$ $Q_3 = \dfrac{13+15}{2} = 14$\\
        $\text{IQR} = 14 - 7 = \mathbf{7}$ hours}

  \item Use the $\mathbf{1.5 \times \text{IQR}}$ \textbf{fence rule} to determine whether
        the value 30 is an outlier. Show both fences.\\[4pt]
        \ans{Lower fence: $7 - 1.5(7) = 7 - 10.5 = -3.5$\\
        Upper fence: $14 + 1.5(7) = 14 + 10.5 = 24.5$\\
        Since $30 > 24.5$, the value 30 \textbf{is an outlier}.}

\end{enumerate}

\vspace{0.18in}

\begin{tcolorbox}[colback=redbg, colframe=redacc, boxrule=0.8pt, arc=2mm,
    left=4mm, right=4mm, top=3mm, bottom=3mm]
\small\textbf{AP-Style Practice --- Sample Response}\\[6pt]
\ans{Because the distribution contains an outlier (30 hours) and is skewed right
(mean $= 11.875 >$ median $= 9.5$), the \textbf{median} is the more appropriate
measure of center; the median of 9.5 hours better represents the typical amount
of sleep for these students than the mean, which is pulled upward by the extreme
value of 30 hours.}

\begin{teachernote}
Award full credit for responses that: (1) identify mean or median by name,
(2) cite skewness or the outlier as justification, and (3) state the value in
context. Do not penalize for ``skewed right'' vs.\ ``has an outlier'' --- either
reason is sufficient.
\end{teachernote}
\end{tcolorbox}
