% ── SECTION 1: MEASURES OF CENTER ────────────────────────────────────────────
\begin{notesbox}{Measures of Center}
\small
\begin{multicols}{2}

\textbf{\textcolor{navy}{Mean} ($\bar{x}$):}
\[\bar{x} = \frac{\sum x_i}{n} = \frac{\blank{2cm}}{\blank{1cm}}\]
\begin{itemize}[itemsep=3pt]
  \item Uses \blank{2.5cm} data values.
  \item \blank{2cm} resistant to outliers.
  \item Pulled toward the \blank{1.5cm} of a skewed distribution.
\end{itemize}

\columnbreak

\textbf{\textcolor{navy}{Median} ($M$):}
\begin{itemize}[itemsep=3pt]
  \item Sort data; take the \blank{2cm} value.
  \item If $n$ is even, average the \blank{2.5cm} two middle values.
  \item \blank{2.2cm} resistant to outliers.
  \item Stays near the \blank{2cm} of the distribution.
\end{itemize}
\end{multicols}
\end{notesbox}

\vspace{0.08in}

% ── SECTION 2: MEASURES OF SPREAD ────────────────────────────────────────────
\begin{notesbox}{Measures of Spread}
\small
\begin{multicols}{2}

\textbf{\textcolor{navy}{Range:}}
\[\text{Range} = \blank{2cm} - \blank{2cm}\]
\begin{itemize}[itemsep=3pt]
  \item Simple but \blank{2cm} resistant.
  \item Only uses the two \blank{2.5cm} values.
\end{itemize}

\vspace{6pt}
\textbf{\textcolor{navy}{Quartiles \& IQR:}}
\begin{itemize}[itemsep=3pt]
  \item $Q_1 = $ median of the \blank{2.5cm} half.
  \item $Q_3 = $ median of the \blank{2.5cm} half.
  \item $\text{IQR} = Q_3 - Q_1 = $ \blank{2cm}
  \item Contains the middle \blank{1cm}\% of data.
  \item \blank{2cm} resistant to outliers.
\end{itemize}

\columnbreak

\textbf{\textcolor{navy}{Standard Deviation} ($s_x$):}
\[s_x = \sqrt{\frac{\sum(x_i - \bar{x})^2}{n-1}}\]
\begin{itemize}[itemsep=3pt]
  \item Typical \blank{2cm} from the mean.
  \item Always $\geq$ \blank{1cm}; units = units of data.
  \item \blank{2cm} resistant to outliers.
  \item Use with \blank{2cm} when distribution is symmetric.
\end{itemize}
\end{multicols}
\end{notesbox}

\vspace{0.08in}

% ── SECTION 3: OUTLIER RULE + DECISION ───────────────────────────────────────
\begin{notesbox}{Outlier Rule \& Choosing Statistics}
\small
\begin{multicols}{2}

\textbf{\textcolor{navy}{$1.5 \times$ IQR Fence Rule:}}
\begin{itemize}[itemsep=4pt]
  \item Lower fence: $Q_1 - 1.5(\text{IQR})$
  \item Upper fence: $Q_3 + 1.5(\text{IQR})$
  \item A value is an outlier if it is \blank{3cm} \\ the lower fence or
        \blank{3cm} the upper fence.
\end{itemize}

\vspace{4pt}
\textbf{Example:} $Q_1 = 19$, $Q_3 = 31$, IQR~$= 12$\\
Lower fence $= \blank{3cm}$\quad Upper fence $= \blank{3cm}$

\columnbreak

\textbf{\textcolor{navy}{Choosing the Right Pair:}}

\vspace{4pt}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\columnwidth}{@{} >{\bfseries}p{0.42\columnwidth} X @{}}
\rowcolor{sky}
\textbf{If shape is\dots} & \textbf{Report\dots} \\
\midrule
Skewed \textit{or} has outliers & \blank{3.5cm} \\
Approximately symmetric & \blank{3.5cm} \\
\end{tabularx}

\vspace{6pt}
\textit{Key:} Mean is pulled toward the \blank{1.5cm}. \\
Median \blank{2.5cm} toward extremes.
\end{multicols}
\end{notesbox}
