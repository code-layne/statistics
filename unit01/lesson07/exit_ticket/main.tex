\small A school nurse recorded the number of hours of sleep reported by
\textbf{8 students} the night before a big exam.
The sorted data are:
\[4,\quad 7,\quad 7,\quad 9,\quad 10,\quad 13,\quad 15,\quad 30\]
\end{tcolorbox}

\vspace{0.2in}

\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item Find the \textbf{mean}. Show your calculation.\\[6pt]
        \writeline

  \item Find the \textbf{median}. Show your work.\\[6pt]
        \writeline

  \item Find $Q_1$, $Q_3$, and the \textbf{IQR}. Show your work.\\[6pt]
        \writeline\\[1pt]\writeline

  \item Use the $\mathbf{1.5 \times IQR}$ \textbf{fence rule} to determine whether
        the value 30 is an outlier. Show both fences.\\[6pt]
        \writeline\\[1pt]\writeline

\end{enumerate}
