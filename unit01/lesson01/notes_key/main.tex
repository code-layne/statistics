% ── VOCABULARY ───────────────────────────────────────────────────────────────
\begin{vocabbox}
\small

\noindent\textbf{\textcolor{navy}{Statistics:}}\quad
\ans{the science of collecting, analyzing, and drawing conclusions from data}\\[4pt]

\noindent\textbf{\textcolor{navy}{Individual:}}\quad
\ans{a single person, object, or event in a dataset}\\[4pt]

\noindent\textbf{\textcolor{navy}{Population:}}\quad
\ans{the entire group of individuals of interest}\\[4pt]

\noindent\textbf{\textcolor{navy}{Sample:}}\quad
\ans{a subset of the population from which data are collected}\\[4pt]

\noindent\textbf{\textcolor{navy}{Census:}}\quad
\ans{data collected from EVERY individual in the population}\\[4pt]

\noindent\textbf{\textcolor{navy}{Variability / Variation:}}\\
\ans{natural differences in values across individuals — the reason statistics exists}\\[4pt]

\noindent\textbf{\textcolor{navy}{Data:}}\quad
\ans{recorded values for individuals; can be categorical or quantitative}\\[4pt]

\noindent\textbf{\textcolor{navy}{Anecdotal evidence:}}\\
\ans{data from a few unrepresentative cases; not reliable for drawing conclusions}\\[4pt]

\end{vocabbox}

\vspace{0.1in}

% ── HOOK ─────────────────────────────────────────────────────────────────────
\begin{hookbox}
\small\textbf{Headline:}\quad
\textit{``Students who sleep more score higher on tests.''}
\hfill\textcolor{slate}{\small(national education study, $n = 12{,}000$)}

\vspace{6pt}
\textbf{Discussion questions — write your thoughts:}
\begin{enumerate}[leftmargin=1.5em, itemsep=4pt]
  \item How do researchers know this? Did they study every student in the world?\\
        \ans{No — they used a sample of 12,000 students and applied statistical
        methods to draw conclusions about a larger population.}

  \item Could this finding be wrong? What would make you doubt it?\\
        \ans{Yes — the sample might not be representative; other factors (e.g., overall
        health, study habits) could explain both sleep and test performance.}

  \item What information would you want before trusting this headline?\\
        \ans{Who was sampled, how sleep was measured, whether other variables
        (confounders) were controlled for, whether the study was an experiment.}
\end{enumerate}
\end{hookbox}

% ── CONCEPT NOTES ─────────────────────────────────────────────────────────────
\begin{notesbox}{Key Concepts}
\small
\begin{multicols}{2}

\textbf{\textcolor{navy}{Population vs. Sample}}\\
\textit{Example:} A researcher surveys 200 adults about exercise habits.
\begin{itemize}[itemsep=2pt]
  \item Individual: \ans{an adult}
  \item Population: \ans{all adults (e.g., in the U.S.)}
  \item Sample: \ans{the 200 randomly chosen adults}
  \item Variable: \ans{exercise habits / days per week}
\end{itemize}

\vspace{4pt}
\textbf{Why use a sample instead of a census?}\\
\ans{A census is too costly, time-consuming, or impossible for large populations.
A well-chosen sample can give reliable estimates.}

\columnbreak

\textbf{\textcolor{navy}{Why Statistics Needs Variation}}\\[3pt]
Complete the statement:\\[4pt]
\textit{``If every individual were identical, we would \ans{NOT}
need statistics, because there would be \ans{nothing}
\ans{different} to explain or measure.''}

% ── GUIDED PRACTICE ──────────────────────────────────────────────────────────
\begin{practicebox}
\small\textbf{Scenario Analysis} \hfill \textcolor{slate}{\scriptsize Sample: Hospital knee-surgery study}

\vspace{5pt}
\textit{Scenario:} \ans{A hospital tracks recovery times for 50 patients after knee surgery.}

\vspace{4pt}
\begin{tabularx}{\linewidth}{@{} l X @{}}
\textbf{Individual:}   & \ans{a single patient who had knee surgery} \\[6pt]
\textbf{Population:}   & \ans{all patients who have knee surgery (at this hospital / in general)} \\[6pt]
\textbf{Sample:}       & \ans{the 50 patients whose recovery times were tracked} \\[6pt]
\textbf{Variable(s):}  & \ans{recovery time (quantitative); could also include pain level, age, etc.} \\[6pt]
\end{tabularx}

\vspace{4pt}
\textbf{Is a census feasible? Why or why not?}\\
\ans{No — tracking every knee-surgery patient would be prohibitively expensive and
time-consuming. The 50-patient sample can still yield useful information.}
