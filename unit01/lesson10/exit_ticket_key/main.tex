\vspace{0.18in}

\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item $z = (\ans{70} - \ans{65})/\ans{2.5} = \ans{+2.0}$\\[6pt]
        Interpretation: \ansline{A height of 70 inches is 2.0 standard deviations above the mean height for adult women.}

  \item $P(\text{height}>70) = P(Z > \ans{2.0}) = 1 - P(Z < \ans{2.0}) = 1 - \ans{0.9772} = \ans{0.0228}$\\[3pt]
        (Also acceptable: Empirical Rule gives $(100\%-95\%)/2 = 2.5\%$.)

  \item $x = 65 + (\ans{-0.67})(2.5) = 65 - 1.675 = \ans{63.3}$ inches

  \item \ansline{$z = (72.5 - 65)/2.5 = 7.5/2.5 = +3.0$. A $z$-score of $+3$ is extremely unusual --- only about 0.15\% of women are taller than 72.5 inches ($P(Z>3) \approx 0.0013$). Yes, this is unusually tall.}

\end{enumerate}

$z(6.4) = (6.4-7.5)/1.1 = -1.0$;\quad $z(8.6) = (8.6-7.5)/1.1 = +1.0$.\\
$P(6.4 < X < 8.6) = P(-1 < Z < 1) = 0.8413 - 0.1587 = \mathbf{0.6826}$
(or by Empirical Rule: $\approx 68\%$).\\[4pt]
\textit{In context:} Approximately 68.3\% of newborns weigh between 6.4 and 8.6 pounds.
This is consistent with the Empirical Rule: the interval is exactly $\mu \pm 1\sigma$.
\end{tcolorbox}
