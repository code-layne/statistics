\small The heights of adult women in the United States are approximately Normally
distributed with mean $\mu = 65$ inches and standard deviation $\sigma = 2.5$ inches.
The pre-drawn Normal curve below represents this distribution.
The shaded region shows heights \textbf{above 70 inches}.

\vspace{0.14in}
\begin{center}
\begin{tikzpicture}[scale=1.0]
  % Shade region z > 2 (x > 70)
  \fill[sky!140, opacity=0.9]
    plot[domain=2:3.5, samples=30] (\x, {2.5*exp(-\x*\x/2)})
    -- (3.5, 0) -- (2, 0) -- cycle;
  % Bell curve
  \draw[thick, navy]
    plot[domain=-3.5:3.5, samples=80] (\x, {2.5*exp(-\x*\x/2)});
  % Axis
  \draw[thick] (-3.8, 0) -- (3.8, 0);
  % Tick marks with height labels
  \foreach \z/\h in {-3/57.5, -2/60, -1/62.5, 0/65, 1/67.5, 2/70, 3/72.5} {
    \draw (\z, -0.12) -- (\z, 0.12);
    \node[below, font=\small] at (\z, -0.15) {\h};
  }
  \node[below, font=\small\itshape] at (0, -0.55) {Height (inches)};
  \node[above, font=\small\bfseries\color{navy}] at (0, 2.52) {$\mu = 65$};
\end{tikzpicture}
\end{center}
\end{tcolorbox}

\vspace{0.18in}

\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item Calculate the $z$-score for a woman who is \textbf{70 inches} tall.
        Show your work and interpret the result in context.\\[6pt]
        $z = \dfrac{x - \mu}{\sigma} = \dfrac{\blank{1.5cm} - \blank{1.5cm}}{\blank{1.5cm}} = \blank{2cm}$\\[6pt]
        Interpretation: \writeline

  \item Use the Normal table or the Empirical Rule to find the proportion of women
        who are \textbf{taller than 70 inches}. Show your work.\\[6pt]
        $P(\text{height} > 70) = P(Z > \blank{2cm}) = 1 - P(Z < \blank{2cm}) = 1 - \blank{2cm} = \blank{2cm}$

  \item Find the height at the \textbf{25th percentile} for adult women.
        Use the inverse Normal: $z^* \approx -0.67$ for the 25th percentile.\\[6pt]
        $x = \mu + z^* \cdot \sigma = 65 + (\blank{2cm})(2.5) = \blank{3cm}$ inches

  \item Is a woman who is 72.5 inches tall unusually tall? Justify with a $z$-score.\\[5pt]
        \writeline\\[1pt]\writeline

\end{enumerate}
