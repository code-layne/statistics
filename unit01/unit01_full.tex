\newcommand{\includefulldoc}[1]{%
  \begingroup
  \subimport{#1/}{main.tex}%
  \endgroup
  \clearpage
}

\newcommand{\includelessonplan}[2][margin=0.6in]{%
  \clearpage
  \newgeometry{#1}%
  \begingroup
  \subimport{#2/}{main.tex}%
  \endgroup
  \clearpage
  \restoregeometry
}

\newcommand{\includepacketpdf}[1]{%
  \IfFileExists{#1}{%
    \includepdf[pages=-,pagecommand={\thispagestyle{plain}}]{#1}%
    \clearpage
  }{}%
}

\newcommand{\slidehandoutnotes}{%
  \begin{minipage}[c][1.9in][t]{\linewidth}
    \small\textbf{Notes:}\par\vspace{0.18in}
    \foreach \i in {1,...,5}{%
      \textcolor{linegray}{\rule{\linewidth}{0.35pt}}\par\vspace{0.24in}%
    }%
  \end{minipage}%
}

\newcommand{\slidehandoutrow}[3]{%
  \ifnum#2>#3
    \multicolumn{2}{c}{}\\[0.12in]
  \else
    \includegraphics[page=#2,width=\linewidth]{#1} &
    \slidehandoutnotes\\[0.12in]
  \fi
}

\newcommand{\includeslidehandout}[2]{%
  \IfFileExists{#1}{%
    \clearpage
    \newgeometry{margin=0.45in}%
    \foreach \first in {1,4,...,#2}{%
      \pgfmathtruncatemacro{\second}{\first+1}%
      \pgfmathtruncatemacro{\third}{\first+2}%
      \noindent
      \begin{tabularx}{\linewidth}{@{}p{0.47\linewidth}@{\hspace{0.35in}}X@{}}
        \slidehandoutrow{#1}{\first}{#2}
        \slidehandoutrow{#1}{\second}{#2}
        \slidehandoutrow{#1}{\third}{#2}
      \end{tabularx}%
      \clearpage
    }%
    \restoregeometry
  }{}%
}
