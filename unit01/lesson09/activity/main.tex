% ── PART 1 ───────────────────────────────────────────────────────────────────
{\large\bfseries\color{navy} Part 1 \quad Compute Five-Number Summaries}

\vspace{0.10in}

\noindent\small A researcher recorded the number of \textbf{hours of sleep per night}
for two groups of high school students.

\vspace{0.10in}

\begin{center}
\small
\renewcommand{\arraystretch}{1.4}
\begin{tabularx}{0.92\linewidth}{@{} l X @{}}
\rowcolor{sky}
\textbf{Athletes} ($n = 15$, sorted): &
6,\enspace 7,\enspace 7,\enspace 8,\enspace 8,\enspace 8,\enspace 9,\enspace
9,\enspace 9,\enspace 10,\enspace 10,\enspace 10,\enspace 11,\enspace 11,\enspace 12 \\
\textbf{Non-Athletes} ($n = 15$, sorted): &
5,\enspace 6,\enspace 6,\enspace 7,\enspace 7,\enspace 7,\enspace 7,\enspace
8,\enspace 8,\enspace 8,\enspace 9,\enspace 9,\enspace 10,\enspace 10,\enspace 14 \\
\end{tabularx}
\end{center}

\vspace{0.14in}

\begin{enumerate}[leftmargin=1.5em, itemsep=8pt]

  \item For $n = 15$, the median is the \blank{2cm} value and $Q_1$ and $Q_3$
        are the medians of the lower and upper \blank{2cm} values.

  \item Complete the five-number summary table for each group.\\[6pt]
        \renewcommand{\arraystretch}{1.6}
        \begin{tabularx}{\linewidth}{@{} l Y Y Y Y Y @{}}
        \rowcolor{sky}
        \textbf{Group} & $\min$ & $Q_1$ & $M$ & $Q_3$ & $\max$ \\
        \midrule
        Athletes      & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} \\
        Non-Athletes  & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} & \blank{1.3cm} \\
        \end{tabularx}

  \item Calculate the IQR for each group, then apply the $1.5 \times \text{IQR}$ fence rule
        for the Non-Athletes.\\[6pt]
        Athletes: IQR $= \blank{2cm}$\quad
        Non-Athletes: IQR $= \blank{2cm}$\\[5pt]
        Non-Athletes upper fence $= Q_3 + 1.5(\text{IQR}) = \blank{6cm}$\\[4pt]
        Is 14 hours an outlier for Non-Athletes? \blank{2cm} \quad Because: \blank{5cm}\\[4pt]
        Last non-outlier for Non-Athletes $= \blank{2cm}$ hours

\end{enumerate}

\vspace{0.10in}

\begin{enumerate}[leftmargin=1.5em, itemsep=8pt, resume]

  \item Read the values from the boxplots to verify your Part 1 calculations.
        Do the displayed values match? \blank{3cm}\\[4pt]
        What does the outlier dot represent in context?\\[5pt]
        \writeline

  \item Approximately what percent of athletes slept between 8 and 10 hours?
        \blank{3cm}\%.\\[4pt]
        Approximately what percent of non-athletes slept between 7 and 9 hours?
        \blank{3cm}\%.

  \item Which group has a higher median sleep time? By how many hours?\\[5pt]
        \writeline

  \item Compare the \textit{spreads} using IQR. What does the difference tell
        you about the consistency of sleep habits in each group?\\[5pt]
        \writeline\\[1pt]\writeline

\end{enumerate}
