\begin{tcolorbox}[colback=sky,colframe=navy,boxrule=0.9pt,arc=2mm,
    left=2mm,right=2mm,top=1.5mm,bottom=1.5mm]
    \textbf{Primary Objective:} Students will classify variables as \textit{categorical} or
    \textit{quantitative}, and further distinguish quantitative variables as \textit{discrete} or
    \textit{continuous}. Students will articulate why variable type determines which statistical
    methods are appropriate (Big Idea: VAR).
\end{tcolorbox}

\begin{skillbox}[Priority Ideas \& Skills]{goldbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP Skill 2 --- Data Analysis}
    \begin{itemize}
        \item Correctly identify whether a variable is categorical or quantitative in context.
        \item Recognize that categorical data produce counts/proportions and quantitative data produce numerical summaries.
        \item Identify the variable type before choosing a display or summary statistic.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Key Understandings}
    \begin{itemize}
        \item A number is not automatically quantitative (e.g., ZIP code, jersey number).
        \item The question ``Does it make sense to average these?'' separates categorical from quantitative.
        \item Discrete vs.\ continuous matters for probability models later (Unit 4).
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[{Vocabulary, Concepts, \& Theorems}]{greenbox}
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Variable} --- a characteristic that can take different values for different individuals
        \item \textbf{Categorical variable} --- places an individual into one of several groups or categories (no meaningful numerical operations)
        \item \textbf{Quantitative variable} --- takes numerical values for which arithmetic makes sense
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Discrete quantitative variable} --- countable values (e.g., number of siblings)
        \item \textbf{Continuous quantitative variable} --- can take any value in an interval (e.g., height, time)
        \item \textbf{Distribution of a variable} --- describes what values the variable takes and how often; concept builds through Lessons 1.3--1.10
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Activate Prior Knowledge \& Spiral Review]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Quick Review (3 min)}\\
    Display the dataset from Lesson 1.1 (e.g., commute times study). Ask: ``Which column records words/labels? Which records numbers? Can you average the labels?'' Use responses to motivate today's classification.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Spiral Connection}
    \begin{itemize}
        \item Lessons 1.3--1.4 display \textit{categorical} variables with tables and graphs.
        \item Lessons 1.5--1.9 display and summarize \textit{quantitative} variables.
        \item Knowing variable type now prevents method-mismatch errors throughout the course.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Hook \& Lesson]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Hook (4--5 min)}\\
    Show a data table with 5 variables about students: name, grade level, GPA, favorite color, hours of sleep. Ask students to sort the variables into two groups without telling them the category names. Debrief: What was the sorting rule? Connect to \textit{categorical} vs.\ \textit{quantitative}.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Lesson Flow (15 min)}
    \begin{enumerate}
        \item Define categorical and quantitative; use the ``average test'' to check.
        \item Work through 6--8 varied examples as a class (include traps: ZIP code, rating on a 1--5 scale).
        \item Introduce discrete vs.\ continuous with physical examples (number of pets vs.\ weight of a pet).
        \item Have students propose one variable for each type from their own lives.
    \end{enumerate}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Explicit Instruction \& Active Monitoring]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Direct Instruction (10 min)}
    \begin{itemize}
        \item Stress the trap variables: area code, student ID, Likert scale ratings. Ask, ``Does averaging jersey numbers give a meaningful result?''
        \item Show that the same \textit{question} can be answered with different variable types (e.g., ``How do students rate the cafeteria?'' as a 1--5 scale \textit{or} as ``satisfied/unsatisfied'').
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Active Monitoring}
    \begin{itemize}
        \item Mini-whiteboard round: display a variable, students hold up C (categorical) or Q (quantitative).
        \item Watch for: students marking ordinal ratings as quantitative without justification.
        \item Prompt: ``Can you find the average? Does that average \textit{mean} something?''
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Group Work \& Differentiation]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Card Sort Activity (8--10 min)}\\
    Groups receive 12 variable cards and sort them into: Categorical / Quantitative Discrete / Quantitative Continuous. After sorting, groups compare with neighbors and resolve disagreements. Debrief 2--3 contested cards whole-class.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Differentiation}
    \begin{itemize}
        \item \textit{Support}: Provide a decision flowchart (``Is it a word/label? $\to$ Categorical. Is it a count? $\to$ Discrete. Otherwise $\to$ Continuous.'')
        \item \textit{Extension}: Students design a 5-variable dataset about a topic of interest and classify each variable, explaining their reasoning in writing.
        \item \textit{Discussion}: Have students debate whether ``number of stars in a review'' is categorical or quantitative.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Individual Work \& Assessment]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Exit Ticket (5 min)}\\
    For each variable, identify type (categorical / quantitative discrete / quantitative continuous):
    \begin{enumerate}
        \item Favorite music genre
        \item Number of songs on a playlist
        \item Duration of the longest song (in minutes)
        \item Country of birth
        \item Temperature at noon today
    \end{enumerate}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP-Style Check}\\
    \textit{``A survey asks respondents for their annual income (in dollars) and their job type (teacher, engineer, other). Which of the following is correct?''} \\[4pt]
    Evaluate students on correct classification AND on providing a brief justification --- AP free-response requires explanation, not just an answer.
\end{minipage}
\end{skillbox}
