\begin{enumerate}[leftmargin=1.5em, itemsep=8pt]

  \item \ans{$\min = 10$, $Q_1 = 18$, $M = 25$, $Q_3 = 34$, last non-outlier $= 55$}\\[4pt]
        IQR $= 34 - 18 = $ \ans{16 minutes}

  \item Upper fence $= 34 + 1.5(16) = 34 + 24 = $ \ans{58 minutes}\\[4pt]
        Is 60 an outlier? \ans{Yes} \quad Because: \ans{$60 > 58$ (upper fence)}

  \item $\approx$ \ans{50\%}. This interval represents \ans{the IQR box --- the
        middle 50\% of delivery times (from $Q_1$ to $Q_3$)}.

  \item \ans{The whisker endpoint at 55 minutes is the \textit{last non-outlier}
        value --- the largest delivery time that does not exceed the upper fence
        of 58 minutes. Because 60 minutes is an outlier (it exceeds 58 minutes),
        it is plotted separately as a dot rather than included in the whisker.}

\end{enumerate}

\begin{tcolorbox}[colback=redbg, colframe=redacc, boxrule=0.8pt, arc=2mm,
    left=4mm, right=4mm, top=3mm, bottom=3mm]
\small\textbf{AP-Style Practice --- Model Response}\\[6pt]
\ans{The distribution of delivery times for the 20 orders is slightly right-skewed.
There is one outlier at 60 minutes, which exceeds the upper fence of 58 minutes
($= 34 + 1.5 \times 16$). The median delivery time was 25 minutes, meaning half
of the orders arrived in fewer than 25 minutes. The middle 50\% of delivery times
ranged from 18 to 34 minutes, giving an IQR of 16 minutes. Excluding the outlier,
the range of delivery times was $55 - 10 = 45$ minutes.}
\end{tcolorbox}
