\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item \ans{Of the 50 students surveyed, the most common preferred study location
        was home, chosen by 44\% of students.}

  \begin{teachernote}
  Must include: total ($n=50$), the variable or context (study location), the
  category (home), and the relative frequency (44\% or 0.44). Deduct for
  missing context or for writing the raw count instead of a proportion/percent.
  \end{teachernote}

  \item \ans{Percentage who do \textit{not} study at home: $100\% - 44\% = 56\%$
        (or $(18+6+4)/50 = 28/50 = 0.56 = 56\%$).}

\end{enumerate}

\textbf{(A)}\quad \ans{$0.42 + 0.18 = 0.60$.
60\% of people in the sample have brown or hazel eyes.}

\vspace{10pt}
\textbf{(B)}\quad \ans{Of the 200 people sampled, brown was the most common eye
color (42\%), followed by blue (28\%), hazel (18\%), and other (12\%). Brown and
hazel together account for 60\% of the sample.}

\begin{teachernote}
For (A): accept $0.60$ or $60\%$; must show the addition.\\
For (B): require at least the most common category with its relative frequency
and context. Full credit requires a description that covers the full distribution.
\end{teachernote}
