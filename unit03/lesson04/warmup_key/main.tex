    \item
    \begin{enumerate}[label=(\alph*), itemsep=4pt]
        \item \ans{The question is leading --- it uses the word ``unfair'' and
        ``needs to be changed,'' pushing respondents toward agreeing. This is
        response bias (question-wording bias).}
        \item \ans{Sample: ``How do you feel about the school's dress code?''
        Strongly support / Support / Neither / Oppose / Strongly oppose.}
    \end{enumerate}

\end{enumerate}
