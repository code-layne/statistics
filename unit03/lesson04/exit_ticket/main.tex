\begin{scenariobox}[Context]{sky}
\small
A researcher mails surveys to 1{,}000 randomly selected adults asking about annual
income. Only 120 return the survey, and most returns are from high-income households.
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]
  \item Was the initial selection of 1{,}000 adults biased? \blank{3cm} Explain.\\
        \writeline

  \item What type of bias affects the 120 responses actually returned?\\
        \blank{8cm}

  \item In which direction does this bias distort the estimate of average income?
        (Circle one: too high / too low) \quad Explain why.\\
        \writeline\\[1pt]\writeline

  \item Suggest one strategy to reduce this bias.\\
        \writeline
\end{enumerate}
\end{practicebox}
