\begin{scenariobox}[Context]{sky}
\small
$X$ and $Y$ are independent random variables representing homework time.
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}lcc@{}}
& \textbf{Mean} & \textbf{Standard Deviation} \\
\hline
Student A ($X$) & 45 min & 8 min \\
Student B ($Y$) & 30 min & 6 min \\
\end{tabular}
\end{center}
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=12pt]
  \item $\mu_T = $ \ans{$45+30=75$ min} \quad
        $\sigma^2_T = $ \ans{$8^2+6^2=64+36=100$} \quad
        $\sigma_T = $ \ans{$\sqrt{100}=10$ min}

  \item $\mu_D = $ \ans{$45-30=15$ min} \quad
        $\sigma^2_D = $ \ans{$64+36=100$} \quad
        $\sigma_D = $ \ans{$10$ min}

  \item $\mu_W = $ \ans{$5+2(45)=95$ min} \quad $\sigma_W = $ \ans{$|2|(8)=16$ min}

  \item \ansline{When you subtract two random variables, either one can be larger or smaller, so the variability increases; we add variances because both random variables contribute uncertainty to the difference.}
\end{enumerate}
\end{practicebox}
