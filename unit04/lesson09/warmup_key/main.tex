{\large\bfseries\color{navy} Spiral Review}

\vspace{0.12in}

\begin{enumerate}

    \item $X$ = complaints per day. $\mu_X = 1.20$, $\sigma_X = 1.17$.
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item \ansline{On average, the store receives about 1.20 complaints per day in the long run.}

        \item \ansline{The number of complaints per day typically varies from the mean of 1.20 by about 1.17 complaints.}
    \end{enumerate}

    \vspace{0.14in}

    \item Goals per game. $\mu_Y = 0.90$, $\sigma_Y \approx 0.700$.
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item $\mu_W = $ \ans{$3 \times 0.90 = 2.70$ points}
        \item $\sigma_W = $ \ans{$3 \times 0.700 = 2.10$ points}
    \end{enumerate}

    \vspace{0.14in}

    \item $a = 3$, $b = 4$.
    \begin{enumerate}[label=(\alph*), itemsep=6pt]
        \item $a + b = $ \ans{7}
        \item $\sqrt{a^2 + b^2} = $ \ans{$\sqrt{9+16} = \sqrt{25} = 5$}
        \item Same? \ans{No.} \ansline{When combining independent RVs, we add variances ($a^2+b^2$) and take the square root --- not add standard deviations directly.}
    \end{enumerate}

\end{enumerate}
