\begin{scenariobox}[Context]{sky}
\small
A spinner is divided into \textbf{8 equal sections} numbered 1 through 8. The spinner
is spun once.
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=8pt]
  \item \ans{$S = \{1,\ 2,\ 3,\ 4,\ 5,\ 6,\ 7,\ 8\}$}

  \item \ans{Even numbers: \{2, 4, 6, 8\} → 4 outcomes out of 8.\\
        $P(\text{even}) = 4/8 = 1/2 = 0.5$.}

  \item \ans{$P(\text{odd}) = 1 - P(\text{even}) = 1 - 0.5 = 0.5$.}

  \item \ans{If this spinner is spun many times, about half of all spins will land on an even number in the long run.}

  \item \ans{$P(\text{number} > 8) = 0$. There is no outcome in the sample space greater than 8; the event is impossible, so its probability is 0.}
\end{enumerate}
\end{practicebox}
