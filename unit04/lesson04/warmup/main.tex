{\large\bfseries\color{navy} Part 1 \quad Spiral Review}

\vspace{0.12in}

\begin{enumerate}

    \item A bag contains 4 red marbles, 3 blue marbles, and 3 green marbles.
    One marble is drawn at random.
    \begin{enumerate}[label=(\alph*), itemsep=6pt]
        \item What is $P(\text{red})$? \blank{4cm}
        \item What is $P(\text{not blue})$? \blank{4cm}
        \item What is $P(\text{red or green})$? \blank{4cm}
    \end{enumerate}

    \vspace{0.18in}

    \item A fair six-sided die is rolled. Let $A = \{\text{roll a number greater than 4}\}$
    and $B = \{\text{roll an even number}\}$.
    \begin{enumerate}[label=(\alph*), itemsep=6pt]
        \item List all outcomes in $A$: \blank{5cm}
        \item List all outcomes in $B$: \blank{5cm}
        \item List any outcomes in \emph{both} $A$ and $B$: \blank{4cm}
        \item Can events $A$ and $B$ occur at the same time? \blank{3cm}
    \end{enumerate}

    \vspace{0.18in}

    \item A student randomly selects a day of the week.
    Let $C = \{\text{weekday}\}$ and $D = \{\text{weekend day}\}$.
    \begin{enumerate}[label=(\alph*), itemsep=6pt]
        \item List all outcomes in $C$: \blank{7cm}
        \item List all outcomes in $D$: \blank{5cm}
        \item Can events $C$ and $D$ both occur? Explain.\\
        \writeline\writeline
    \end{enumerate}

\end{enumerate}

% ── SECTION 2: Looking Ahead ──────────────────────────────────────────────────
{\large\bfseries\color{navy} Part 2 \quad Looking Ahead}

\vspace{0.12in}

\noindent Today we study events that \textbf{cannot both occur at the same time}.

\vspace{0.14in}

\begin{tcolorbox}[colback=greenbg, colframe=greenacc, boxrule=0.8pt, arc=2mm,
    left=4mm, right=4mm, top=3mm, bottom=3mm]
    \small
    \begin{itemize}[leftmargin=1.3em, itemsep=8pt]
      \item Two events are \textbf{mutually exclusive} if they share no outcomes.
      \item The key check: does any single outcome belong to \emph{both} events?
      \item If yes --- not mutually exclusive. If no --- they are mutually exclusive.
    \end{itemize}
\end{tcolorbox}
