\begin{scenariobox}[Context]{sky}
\small
A manufacturer claims only 3\% of its products are defective. A store orders
200 items. Assume the items are independent and the order is less than 10\%
of all products ever made. Let $X =$ the number of defective items.
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=10pt]

  \item State the distribution of $X$: $X \sim$ \blank{4cm}

  \item Calculate $\mu_X$ and $\sigma_X$. Show the formula setup.

  \vspace{4pt}
  $\mu_X = $ \blank{6cm} $= $ \blank{2cm}

  $\sigma_X = $ \blank{6cm} $\approx $ \blank{2cm}

  \item Interpret $\mu_X$ in context (write a complete sentence).

  \vspace{4pt}
  \writeline\\[1pt]\writeline

  \item Find $P(X \le 10)$. Write the calculator notation and the answer.

  \vspace{4pt}
  $P(X \le 10) = $ \blank{6cm} $\approx $ \blank{2cm}

\end{enumerate}
\end{practicebox}
