\textbf{1.} Tomato seeds: $n=30$, $p=0.85$.
\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=5pt]
  \item \ans{$\mu_X = 30(0.85) = 25.5$; \quad $\sigma_X = \sqrt{30(0.85)(0.15)} = \sqrt{3.825} \approx 1.96$}
  \item \ans{If a gardener plants 30 seeds many times, an average of 25.5 seeds per planting would germinate.}
  \item \ans{$P(X \le 22) = \texttt{binomcdf(30, 0.85, 22)} \approx 0.0480$}
  \item \ans{$P(X \ge 28) = 1 - P(X \le 27) = 1 - \texttt{binomcdf(30, 0.85, 27)} \approx 1 - 0.5888 = 0.4112$}
\end{enumerate}

\vspace{6pt}
\textbf{2.} Multiple choice guessing: $n=50$, $p=0.20$.
\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=5pt]
  \item \ans{$X \sim B(50, 0.20)$}
  \item \ans{$\mu_X = 50(0.20) = 10$; \quad $\sigma_X = \sqrt{50(0.20)(0.80)} = \sqrt{8} \approx 2.83$}
  \item \ans{$P(X \ge 30) = 1 - \texttt{binomcdf(50, 0.20, 29)} \approx 1 - 1.000 \approx 0.0000$. It is essentially impossible to pass by guessing --- the mean is only 10 and 30 is about 7 standard deviations above the mean.}
\end{enumerate}

\vspace{6pt}
\textbf{3.} ER discharges: $n=120$, $p=0.40$.
\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=5pt]
  \item \ans{$\mu_X = 120(0.40) = 48$; \quad $\sigma_X = \sqrt{120(0.40)(0.60)} = \sqrt{28.8} \approx 5.37$}
  \item \ans{On days with 120 ER visits, the hospital would average 48 patients discharged within 4 hours over many such days.}
  \item \ans{$P(40 \le X \le 50) = \texttt{binomcdf(120, 0.40, 50)} - \texttt{binomcdf(120, 0.40, 39)} \approx 0.6473 - 0.0827 = 0.5646$}
\end{enumerate}

\vspace{6pt}
\textbf{4.} Public transit support: $n=45$, $p=0.62$.
\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=5pt]
  \item \ans{$X \sim B(45, 0.62)$. B: each adult either supports (yes) or not (no). I: sample is $< 10\%$ of the population. N: fixed $n=45$. S: same $p=0.62$.}
  \item \ans{$\mu_X = 45(0.62) = 27.9$. In many samples of 45 adults from this city, the average number who support the plan would be 27.9.}
  \item \ans{$\sigma_X = \sqrt{45(0.62)(0.38)} = \sqrt{10.602} \approx 3.26$}
  \item \ans{$P(X \ge 30) = 1 - P(X \le 29) = 1 - \texttt{binomcdf(45, 0.62, 29)} \approx 1 - 0.6614 = 0.3386$}
\end{enumerate}

\vspace{6pt}
\textbf{5. Extension:}
\ans{(i) $P(X \ge 1) = 1 - P(X=0) = 1 - \binom{20}{0}(0.05)^0(0.95)^{20} = 1 - (0.95)^{20} \approx 1 - 0.3585 = 0.6415$. (ii) $1 - \texttt{binomcdf(20, 0.05, 0)} \approx 0.6415$. They agree because $\texttt{binomcdf(n, p, 0)} = P(X \le 0) = P(X=0)$, so both expressions equal $1 - P(X=0)$.}
