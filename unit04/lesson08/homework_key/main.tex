\begin{practicebox}
\small

\textbf{1.} Customer complaints.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=6pt]
  \item
\renewcommand{\arraystretch}{1.9}
\begin{tabularx}{\linewidth}{@{} c c X @{}}
$x$ & $P(X=x)$ & $x \cdot P(X=x)$ \\
\hline
0 & 0.35 & \ans{0.00} \\
1 & 0.30 & \ans{0.30} \\
2 & 0.20 & \ans{0.40} \\
3 & 0.10 & \ans{0.30} \\
4 & 0.05 & \ans{0.20} \\
\hline
& & $\mu_X =$ \ans{1.20} \\
\end{tabularx}

  \item \ansline{On average, the store receives about 1.20 complaints per day in the long run.}

  \item \ans{$\sigma^2_X=0.35(0-1.2)^2+0.30(1-1.2)^2+0.20(2-1.2)^2+0.10(3-1.2)^2+0.05(4-1.2)^2$}\\
        \ans{$= 0.504+0.012+0.128+0.324+0.392 = 1.360$; $\sigma_X=\sqrt{1.360}\approx 1.17$ complaints}

  \item \ansline{The number of complaints per day typically varies from the mean of 1.20 by about 1.17 complaints.}
\end{enumerate}

\vspace{0.14in}

\textbf{2.} Insurance payout.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=8pt]
  \item \ans{$P(X=0)=0.96$; $P(X=\$10{,}000)=0.04$}

  \item $\mu_X = $ \ans{$\$0(0.96)+\$10{,}000(0.04) = \$400$}

  \item \ansline{On average, the company expects to pay out \$400 per policy per year in the long run.}
\end{enumerate}

\vspace{0.14in}

\textbf{3.} Interpretation practice.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=8pt]
  \item \ansline{On average, a student sleeps about 4.5 hours per school night in the long run.}

  \item \ansline{The number of hours a student sleeps typically varies from the mean of 4.5 hours by about 2.1 hours.}
\end{enumerate}

\end{practicebox}

\begin{extensionbox}
\small
\textbf{Extension.}
\ansline{Let $p = P(\text{win})$. Fair game: $\mu = 0$, so $\$6p + \$0(1-p) - \$2 = 0 \Rightarrow 6p=2 \Rightarrow p=1/3 \approx 0.333$.}
\end{extensionbox}
