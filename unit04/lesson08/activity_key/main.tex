\begin{tcolorbox}[colback=greenbg!60, colframe=greenacc, title=\textbf{Tier E --- Extension},
    arc=2mm, boxrule=0.8pt, left=4mm, right=4mm, top=3mm, bottom=3mm]
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=8pt]
  \item \ans{$\mu_X = \$0(0.60)+\$3(0.30)+\$10(0.10) = \$0+\$0.90+\$1.00=\$1.90$}\\
        \ans{$\mu_Y = \$0(0.40)+\$2(0.40)+\$6(0.20) = \$0+\$0.80+\$1.20=\$2.00$}

  \item \ans{$\sigma_X \approx \$2.84$; \quad $\sigma_Y \approx \$2.19$}

        \begin{teachernote}
        Game A: $\sigma^2_X=0.60(0-1.90)^2+0.30(3-1.90)^2+0.10(10-1.90)^2 \approx 8.09$; $\sigma_X \approx 2.84$.
        Game B: $\sigma^2_Y=0.40(0-2.00)^2+0.40(2-2.00)^2+0.20(6-2.00)^2=1.60+0+3.20=4.80$; $\sigma_Y \approx 2.19$.
        \end{teachernote}

  \item \ansline{Game B has the higher expected prize (\$2.00 vs.\ \$1.90).}

  \item \ansline{Game A is riskier ($\sigma_X \approx \$2.84 > \sigma_Y \approx \$2.19$).}

  \item \ansline{Net winnings A: $\$1.90 - \$3 = -\$1.10$ per play. Net winnings B: $\$2.00-\$3=-\$1.00$ per play.}

  \item \ansline{Game B is slightly better: higher expected net winnings ($-\$1.00$) and lower risk. However,}\\
        \ansline{both games have negative expected net winnings, so neither is a good long-run bet.}
\end{enumerate}
\end{tcolorbox}
