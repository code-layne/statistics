\begin{objectivebox}
\small By the end of this lesson, I will be able to\dots\\[4pt]
\ans{Compute the expected value of a discrete random variable using $\mu_X = \sum x \cdot P(X=x)$.}\\[1pt]
\ans{Compute the standard deviation using $\sigma_X = \sqrt{\sum (x-\mu_X)^2 \cdot P(X=x)}$.}\\[1pt]
\ans{Interpret $\mu_X$ and $\sigma_X$ in context.}
\end{objectivebox}

\vspace{0.1in}

\begin{vocabbox}
\termblanklong{Expected value ($\mu_X$)} \ans{The long-run average of a random variable; a weighted average of all possible values.}
\termblanklong{Variance ($\sigma^2_X$)} \ans{The average squared deviation from $\mu_X$, weighted by probabilities: $\sum (x-\mu_X)^2 P(X=x)$.}
\termblanklong{Standard deviation ($\sigma_X$)} \ans{$\sqrt{\sigma^2_X}$; measures typical deviation from the mean in original units.}
\termblanklong{Long-run average} \ans{The value the sample mean approaches over many independent repetitions of the random process.}
\end{vocabbox}

\vspace{0.1in}

\begin{hookbox}
\small
A carnival game costs \$2 to play. Prize distribution:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}ccc@{}}
Prize ($x$) & \$0 & \$5 \\
\hline
Probability & 0.70 & 0.30 \\
\end{tabular}
\end{center}
Expected prize per play: \ans{$\$0(0.70)+\$5(0.30) = \$1.50$. In 100 plays, expect $\approx\$150$.}\\
Is playing worth \$2? \ans{No --- the expected prize (\$1.50) is less than the cost (\$2.00).}
\end{hookbox}

\vspace{0.1in}

\begin{notesbox}{Mean (Expected Value) of a Random Variable}
\small

The \textbf{expected value} of $X$ is a \ans{weighted} average of all possible values.

\vspace{8pt}
\begin{center}
\TallMath{\mu_X = \sum x \cdot P(X=x) = x_1 \cdot P(X=x_1) + x_2 \cdot P(X=x_2) + \cdots}
\end{center}

\vspace{8pt}
\textbf{Computation Table Example:}

\vspace{4pt}
\renewcommand{\arraystretch}{1.8}
\begin{tabularx}{\linewidth}{@{} c c X @{}}
$x$ & $P(X=x)$ & $x \cdot P(X=x)$ \\
\hline
\$0 & 0.70 & \ans{$\$0 \times 0.70 = \$0.00$} \\
\$5 & 0.30 & \ans{$\$5 \times 0.30 = \$1.50$} \\
\end{tabularx}

\vspace{6pt}
$\mu_X = $ \ans{$\$1.50$}

\vspace{6pt}
\textbf{Interpret:} \ans{long run} means that over many plays, the average prize per play is approximately \ans{\$1.50}.

\end{notesbox}

\vspace{0.1in}

\begin{notesbox}{Standard Deviation of a Random Variable}
\small

The \textbf{standard deviation} is $\sigma_X = $ \ans{$\sqrt{\sigma^2_X}$}.

\vspace{8pt}
\textbf{Full Computation Table:} $X$ = cars per household ($\mu_X = 1.55$).

\vspace{4pt}
\renewcommand{\arraystretch}{2.0}
\begin{tabularx}{\linewidth}{@{} c c c c X @{}}
$x$ & $P(X=x)$ & $x \cdot P(X=x)$ & $(x - \mu_X)^2$ & $(x-\mu_X)^2 \cdot P(X=x)$ \\
\hline
0 & 0.10 & \ans{0.00} & \ans{2.4025} & \ans{0.2403} \\
1 & 0.40 & \ans{0.40} & \ans{0.3025} & \ans{0.1210} \\
2 & 0.35 & \ans{0.70} & \ans{0.2025} & \ans{0.0709} \\
3 & 0.15 & \ans{0.45} & \ans{2.1025} & \ans{0.3154} \\
\hline
\textbf{Sum} & & $\mu_X =$ \ans{1.55} & & $\sigma^2_X =$ \ans{0.7476} \\
\end{tabularx}

\vspace{6pt}
$\sigma_X = $ \ans{$\sqrt{0.7476} \approx 0.865$ cars}

\vspace{6pt}
\textbf{Interpret:} \ansline{The number of cars per household typically varies from the mean of 1.55 cars by about 0.87 cars.}

\end{notesbox}

\vspace{0.1in}

\begin{practicebox}
\small
$X$ = number of goals per game.

\renewcommand{\arraystretch}{2.0}
\begin{tabularx}{\linewidth}{@{} c c c c X @{}}
$x$ & $P(X=x)$ & $x \cdot P(X=x)$ & $(x - \mu_X)^2$ & $(x-\mu_X)^2 \cdot P(X=x)$ \\
\hline
0 & 0.30 & \ans{0.00} & \ans{$(0-0.90)^2=0.81$} & \ans{0.243} \\
1 & 0.50 & \ans{0.50} & \ans{$(1-0.90)^2=0.01$} & \ans{0.005} \\
2 & 0.20 & \ans{0.40} & \ans{$(2-0.90)^2=1.21$} & \ans{0.242} \\
\hline
\textbf{Sum} & & $\mu_X =$ \ans{0.90} & & $\sigma^2_X =$ \ans{0.490} \\
\end{tabularx}

\vspace{6pt}
$\sigma_X = $ \ans{$\sqrt{0.490} \approx 0.700$ goals}

\vspace{6pt}
\ansline{$\mu_X$: On average, this team scores 0.90 goals per game in the long run.}

\vspace{2pt}
\ansline{$\sigma_X$: The number of goals per game typically varies from 0.90 by about 0.70 goals.}
\end{practicebox}
