% ── SECTION 2: Looking Ahead ──────────────────────────────────────────────────
{\large\bfseries\color{navy} Part 2 \quad Looking Ahead}

\vspace{0.12in}

\noindent Today we formalize two ideas: \textbf{independence} and \textbf{unions} of events.

\vspace{0.14in}

\begin{tcolorbox}[colback=greenbg, colframe=greenacc, boxrule=0.8pt, arc=2mm,
    left=4mm, right=4mm, top=3mm, bottom=3mm]
    \small
    \begin{itemize}[leftmargin=1.3em, itemsep=8pt]
      \item Two events are \textbf{independent} if $P(A|B) = P(A)$ (knowing $B$ doesn't change $A$'s probability).
      \item The probability of a \textbf{union} is: $P(A \cup B) = P(A) + P(B) - P(A \cap B)$.
      \item \textbf{Independent $\neq$ Mutually exclusive.} These are different concepts!
    \end{itemize}
\end{tcolorbox}
