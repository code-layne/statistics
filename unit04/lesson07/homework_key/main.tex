\begin{practicebox}
\small

\textbf{1.} A spinner numbered 1--5.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=8pt]
  \item \ans{Each $P(X=x) = 0.20$}

  \item \ansline{All probabilities $= 0.20 \ge 0$; sum $= 5 \times 0.20 = 1.00$. Valid. \checkmark}

  \item $P(X \ge 3) = $ \ans{$0.20+0.20+0.20=0.60$}

  \item $P(2 \le X \le 4) = $ \ans{$0.20+0.20+0.20=0.60$}
\end{enumerate}

\vspace{0.14in}

\textbf{2.} Broken eggs.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=8pt]
  \item $P(Y=3) = $ \ans{$1 - (0.68+0.17+0.09+0.02) = 0.04$}
  \item $P(Y \le 1) = $ \ans{$0.68+0.17=0.85$}
  \item $P(Y \ge 1) = $ \ans{$1-0.68=0.32$}
  \item \ansline{There is a 68\% probability that a randomly selected carton contains no broken eggs.}
\end{enumerate}

\vspace{0.14in}

\textbf{3.} Valid distribution(s): \ans{(B) only.}

\ansline{(A) invalid: $0.30+0.50+0.40=1.20 \ne 1$. (B) valid: all $\ge 0$, sum $= 1.00$. (C) invalid: $P(X=1)=-0.10 < 0$.}

\vspace{0.14in}

\textbf{4.} Homework hours.

\renewcommand{\arraystretch}{1.6}
\begin{tabularx}{\linewidth}{@{} l X X X X @{}}
$x$ & 0 & 1 & 2 & 3 \\
\hline
$P(X \le x)$ & \ans{0.05} & \ans{0.35} & \ans{0.80} & \ans{1.00} \\
\end{tabularx}

\vspace{6pt}
$P(X \le 2) = $ \ans{0.80} \quad \ansline{There is an 80\% probability that a student spends 2 or fewer hours on homework per night.}

\end{practicebox}
