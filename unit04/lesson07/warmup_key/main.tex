{\large\bfseries\color{navy} Spiral Review}

\vspace{0.12in}

\begin{enumerate}

    \item A fair six-sided die is rolled once. Let $A$ = ``rolling an even number'' and
          $B$ = ``rolling a number greater than 4.''

    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item List the sample space $S$. \quad \ans{$\{1, 2, 3, 4, 5, 6\}$}
        \item Find $P(A)$. \quad \ans{$3/6 = 1/2$}
        \item Find $P(B)$. \quad \ans{$2/6 = 1/3$}
        \item Are $A$ and $B$ mutually exclusive? Justify your answer.\\
              \ansline{No. $A \cap B = \{6\}$, which is non-empty, so they can occur together.}
    \end{enumerate}

    \vspace{0.14in}

    \item A bag contains 3 red, 2 blue, and 5 green marbles. A marble is drawn at random.
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item $P(\text{red}) = $ \ans{$3/10 = 0.30$}
        \item $P(\text{not green}) = $ \ans{$5/10 = 0.50$}
        \item $P(\text{red or blue}) = $ \ans{$5/10 = 0.50$}
    \end{enumerate}

    \vspace{0.14in}

    \item The table below shows the results of a survey of 100 students.

    \vspace{6pt}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{@{}lcc@{}}
    & \textbf{Studied} & \textbf{Did Not Study} \\
    \hline
    \textbf{Passed} & 55 & 15 \\
    \textbf{Failed}  & 10 & 20 \\
    \end{tabular}

    \vspace{6pt}
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item $P(\text{studied and passed}) = $ \ans{$55/100 = 0.55$}
        \item $P(\text{passed} \mid \text{studied}) = $ \ans{$55/65 \approx 0.846$}
    \end{enumerate}

    \vspace{0.14in}

    \item \ansline{$P(\text{both bus}) = \dfrac{12}{30} \cdot \dfrac{11}{29} = \dfrac{132}{870} \approx 0.152$}

\end{enumerate}
