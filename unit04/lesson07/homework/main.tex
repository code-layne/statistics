\textbf{1.} A spinner is divided into 5 equal sections numbered 1--5. Let $X$ be the number
the spinner lands on.

\begin{enumerate}[label=(\alph*), leftmargin=2em, itemsep=8pt]
  \item Write the probability distribution for $X$ as a table.

        \vspace{0.35in}

  \item Verify the two conditions for a valid probability distribution.\\
        \writeline

  \item Find $P(X \ge 3)$.  \quad \blank{4cm}

  \item Find $P(2 \le X \le 4)$. \quad \blank{4cm}
\end{enumerate}
