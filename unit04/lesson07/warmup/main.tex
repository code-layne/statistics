{\large\bfseries\color{navy} Spiral Review}

\vspace{0.12in}

\begin{enumerate}

    \item A fair six-sided die is rolled once. Let $A$ = ``rolling an even number'' and
          $B$ = ``rolling a number greater than 4.''

    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item List the sample space $S$. \quad \blank{7cm}
        \item Find $P(A)$. \quad \blank{4cm}
        \item Find $P(B)$. \quad \blank{4cm}
        \item Are $A$ and $B$ mutually exclusive? Justify your answer.\\
              \writeline
    \end{enumerate}

    \vspace{0.14in}

    \item A bag contains 3 red, 2 blue, and 5 green marbles. A marble is drawn at random.
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item $P(\text{red}) = $ \blank{2.5cm}
        \item $P(\text{not green}) = $ \blank{2.5cm}
        \item $P(\text{red or blue}) = $ \blank{2.5cm}
    \end{enumerate}

    \vspace{0.14in}

    \item The table below shows the results of a survey of 100 students about whether
          they studied the night before a test.

    \vspace{6pt}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{@{}lcc@{}}
    & \textbf{Studied} & \textbf{Did Not Study} \\
    \hline
    \textbf{Passed} & 55 & 15 \\
    \textbf{Failed}  & 10 & 20 \\
    \end{tabular}

    \vspace{6pt}
    \begin{enumerate}[label=(\alph*), itemsep=8pt]
        \item If a student is chosen at random, find $P(\text{studied and passed})$.
              \quad \blank{3cm}
        \item Find $P(\text{passed} \mid \text{studied})$. \quad \blank{3cm}
    \end{enumerate}

    \vspace{0.14in}

    \item A class has 12 students who take the bus and 18 who don't. If two students
          are selected at random \emph{without} replacement, what is the probability
          both take the bus? Show your work.\\[4pt]
          \writeline\\[1pt]\writeline

\end{enumerate}
