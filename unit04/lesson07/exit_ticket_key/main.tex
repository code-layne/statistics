\begin{scenariobox}[Context]{sky}
\small
$X$ = number of siblings for a randomly chosen student at Jefferson High School.
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}ccccc@{}}
$x$ & 0 & 1 & 2 & 3 \\
\hline
$P(X=x)$ & 0.15 & 0.45 & 0.30 & 0.10 \\
\end{tabular}
\end{center}
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=12pt]
  \item \ansline{(1) All probabilities $\ge 0$: yes. (2) $0.15+0.45+0.30+0.10=1.00$ \checkmark. Valid.}

  \item \ansline{$P(X \ge 2) = P(2)+P(3) = 0.30+0.10 = 0.40$}

  \item \ansline{The distribution is right-skewed. Most students have 1 sibling (45\%), and having}\\
        \ansline{3 siblings is least common (10\%). Fewer than half of students have 2 or more siblings.}
\end{enumerate}
\end{practicebox}
