\begin{scenariobox}[Context]{sky}
\small
A nutritionist studied the relationship between diet type and a health outcome
for 200 adults. The results are shown below.

\vspace{6pt}
\renewcommand{\arraystretch}{1.4}
\begin{tabularx}{\linewidth}{@{} l X X X r @{}}
 & \textbf{Plant-based} & \textbf{Omnivore} & \textbf{Keto} & \textbf{Total} \\
\hline
\textbf{Healthy outcome}  & 50 & 40 & 30 & 120 \\
\textbf{At-risk outcome}  & 10 & 40 & 30 & 80 \\
\hline
\textbf{Total}            & 60 & 80 & 60 & 200 \\
\end{tabularx}

\vspace{4pt}
One adult is selected at random.
\end{scenariobox}

\vspace{0.1in}

\begin{practicebox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=14pt]
  \item Find $P(\text{Healthy outcome} | \text{Plant-based})$. Write the formula,
        substitute values, and simplify.

        \vspace{4pt}
        $P(A|B) = \dfrac{P(A \cap B)}{P(B)} = \dfrac{\blank{2cm}}{\blank{2cm}} = $ \blank{3cm}

        Interpret in context: \writeline\writeline

  \item Apply the multiplication rule to find $P(\text{Keto} \cap \text{At-risk})$.
        Use $P(\text{Keto})$ and $P(\text{At-risk}|\text{Keto})$.

        \vspace{4pt}
        $P(\text{Keto}) = $ \blank{3cm} \quad $P(\text{At-risk}|\text{Keto}) = $ \blank{3cm}\\[4pt]
        $P(\text{Keto} \cap \text{At-risk}) = $ \blank{3cm} $\times$ \blank{3cm} $=$ \blank{3cm}\\[4pt]
        Check against the table: \blank{4cm}

  \item A researcher claims: ``The probability of a healthy outcome given a plant-based
        diet is equal to the probability of a healthy outcome given an omnivore diet.''
        Use calculations to assess this claim.\\
        \writeline\\[1pt]\writeline\\[1pt]\writeline
\end{enumerate}
\end{practicebox}
