          \item Random: \writeline
          \item Normal: \writeline
        \end{itemize}
  \item \textbf{Do:} $SE = \dfrac{s}{\sqrt{n}} = $ \blank{3cm} \quad
        $t = \dfrac{\bar x - \mu_0}{SE} = $ \blank{4cm} $\approx$ \blank{2cm}

        $p$-value: \texttt{tcdf}(\blank{2cm}, $10^{99}$, \blank{2cm}) $\approx$ \blank{2cm}

        \writeline
