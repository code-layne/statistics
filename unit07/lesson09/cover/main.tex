\begin{learningtargetbox}
\small
\begin{enumerate}[leftmargin=1.5em, itemsep=5pt]
  \item I can calculate the two-sample $t$-test statistic
        $t = \dfrac{(\bar x_1 - \bar x_2) - 0}{\sqrt{s_1^2/n_1 + s_2^2/n_2}}$
        and find the associated $p$-value using technology.
        \textit{(VAR-7.I, Skill 3.E)}
  \item I can interpret the $p$-value of a two-sample $t$-test in context, recognizing
        that it is computed assuming the two population means are equal ($H_0$ is true).
        \textit{(DAT-3.G, Skill 4.B)}
  \item I can make and justify a formal conclusion by comparing the $p$-value to $\alpha$
        and restating the decision in the context of the research question.
        \textit{(DAT-3.H, Skill 4.E)}
\end{enumerate}
\end{learningtargetbox}
