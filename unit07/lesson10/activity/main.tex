\begin{scenariobox}[Scenario 2 --- Vaccine Side Effects]{sky}
A pharmaceutical company randomly samples 200 participants from a clinical trial.
Of 100 who received the vaccine, 18 reported a side effect. Of 100 who received a
placebo, 9 reported a side effect. The company wants to estimate how much higher the
proportion with side effects is for the vaccine group compared to the placebo group.

\textbf{Tier R:} Select the procedure and list the conditions.
\begin{itemize}
  \item Procedure: \blank{8cm}
  \item Large Counts (vaccine): $n_1 \hat p_1 =$ \blank{2.5cm} and $n_1(1-\hat p_1) =$ \blank{2.5cm}
  \item Large Counts (placebo): $n_2 \hat p_2 =$ \blank{2.5cm} and $n_2(1-\hat p_2) =$ \blank{2.5cm}
\end{itemize}

\textbf{Tier A:} Construct a 95\% confidence interval.
\begin{enumerate}
  \item \textbf{State:} Procedure: \blank{7cm} \quad
        Parameter: $p_1 - p_2 =$ \blank{6cm}
  \item \textbf{Plan:} Verify conditions (with numbers):
        \begin{itemize}
          \item Random: \writeline
          \item 10\%: \writeline
          \item Large Counts: \writeline
        \end{itemize}
  \item \textbf{Do:} $\hat p_1 =$ \blank{1.5cm} \; $\hat p_2 =$ \blank{1.5cm} \;
        $SE = \sqrt{\dfrac{\hat p_1(1-\hat p_1)}{100} + \dfrac{\hat p_2(1-\hat p_2)}{100}} \approx$ \blank{1.8cm}

        Interval: $(0.18-0.09) \pm 1.96 \times$ \blank{1.8cm} $= ($ \blank{1.5cm}, \blank{1.5cm} $)$
  \item \textbf{Conclude:} \writeline
\end{enumerate}
\end{scenariobox}
