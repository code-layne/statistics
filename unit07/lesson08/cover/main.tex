\begin{learningtargetbox}

\begin{enumerate}[leftmargin=1.5em, itemsep=5pt]
  \item I can identify when to use a two-sample $t$-test for a difference of two
        population means, distinguishing it from a matched-pairs $t$-test.
        \textit{(VAR-7.F, Skill 1.E)}
  \item I can state null and alternative hypotheses for a two-sample $t$-test in
        terms of $\mu_1 - \mu_2$, choosing the correct direction for $H_a$.
        \textit{(VAR-7.G, Skill 1.F)}
  \item I can verify the Random/Independence and Normal conditions for both samples
        in a significance test for a difference of two population means.
        \textit{(VAR-7.H, Skill 4.C)}
\end{enumerate}
