% ── OBJECTIVE ────────────────────────────────────────────────────────────────
\begin{objectivebox}
\small By the end of this lesson, I will be able to\dots\\[4pt]
\ansline{identify nonlinear patterns in scatterplots and residual plots.}\\[1pt]
\ansline{apply log transformations to linearize exponential and power data.}\\[1pt]
\ansline{write and interpret the regression equation in transformed form.}\\[1pt]
\ansline{back-transform to make predictions on the original scale.}
\end{objectivebox}

\vspace{0.1in}

% ── VOCABULARY ───────────────────────────────────────────────────────────────
\begin{vocabbox}
\termblanklong{Nonlinear relationship}\ans{a pattern better described by a curve than a line}
\termblanklong{Transformation}\ans{a mathematical operation (e.g., $\log$, $\sqrt{\phantom{y}}$) applied to change the scale of a variable}
\termblanklong{Exponential model ($y = a \cdot b^x$)}\ans{linearized by regressing $\log y$ on $x$}
\termblanklong{Power model ($y = a\,x^b$)}\ans{linearized by regressing $\log y$ on $\log x$}
\termblanklong{Back-transformation}\ans{reversing the log to return to original units: $\hat{y} = 10^{\widehat{\log y}}$}
\termblanklong{Linearization check}\ans{inspect residual plot of transformed model for random scatter}
\end{vocabbox}

\vspace{0.1in}

% ── HOOK ─────────────────────────────────────────────────────────────────────
\begin{hookbox}
\small\textbf{Opening Question:} A scatterplot of bacteria count (millions) vs.\ hours of
growth curves upward sharply. A student fits a linear regression and gets a curved residual
plot. What does the residual plot tell us? What should we do next?

\vspace{6pt}
What the residual plot tells us: \ans{the linear model is not appropriate; the relationship is nonlinear}

\vspace{4pt}
\textbf{Key idea:} Instead of abandoning regression, we can \textbf{transform} the data to
make the relationship \ans{linear}, then apply the linear regression tools we already know.
\end{hookbox}

\vspace{0.1in}

% ── STEP 1 ───────────────────────────────────────────────────────────────────
\begin{notesbox}{Step 1 --- Diagnosing Nonlinearity (DAT-1.G)}
\small

\noindent A linear model is \textbf{not appropriate} when:
\begin{itemize}[itemsep=2pt]
  \item The \textbf{scatterplot} shows a \ans{curved} pattern (not a straight line), \textbf{or}
  \item The \textbf{residual plot} shows a \ans{curved/systematic} pattern (not random scatter).
\end{itemize}

\vspace{6pt}
\textbf{Example:} Bacteria count (millions) $y$ vs.\ hours $x$. \\[4pt]
\noindent Original scatterplot: \ans{curves upward (exponential shape)}.
Residual plot: \ans{shows a curved/arch pattern}.

\vspace{6pt}
\textbf{Conclusion:} The linear model is \ans{not} appropriate because \ans{the residual
plot shows a systematic curved pattern, not random scatter}.

\end{notesbox}

\vspace{0.1in}

% ── STEP 2 ───────────────────────────────────────────────────────────────────
\begin{notesbox}{Step 2 --- The Exponential Model: Transform $y$ (DAT-1.H)}
\small

\noindent \textbf{When to use:} The scatterplot looks like an exponential curve
($y$ grows or decays multiplicatively).

\vspace{6pt}
\textbf{Transformation:} Apply $\log_{10}$ to the \ans{$y$} values only.

\vspace{4pt}
\textbf{Why it works:}
\[
  y = a \cdot b^x
  \;\;\Longrightarrow\;\;
  \log y = \log a + (\log b)\,x
  \;\;\Longrightarrow\;\;
  \underbrace{\log y}_{\text{new } y'} = \underbrace{\log a}_{A} + \underbrace{(\log b)}_{B}\,x
\]
This is the equation of a \ans{straight line (linear equation)} in $(\log y,\, x)$.

\vspace{8pt}
\textbf{Regression of $\log y$ on $x$} gives: \quad $\widehat{\log y} = $ \ans{$a + bx$ (with $a$ and $b$ from technology)}

\vspace{6pt}
\textbf{Back-transform} to predict on the original scale:
\[
  \hat{y} = 10^{\widehat{\log y}} = 10^{a + bx}
\]

\vspace{6pt}
\textbf{Example} --- bacteria count data. Technology gives: $\widehat{\log(\text{count})} = 0.85 + 0.22x$

\vspace{4pt}
Write the back-transformed exponential equation:
\ans{$\widehat{\text{count}} = 10^{0.85 + 0.22x} = 10^{0.85} \cdot (10^{0.22})^x \approx 7.08 \cdot 1.66^x$}

\vspace{4pt}
Predict count at $x = 6$ hours:
\ans{$\widehat{\log(\text{count})} = 0.85 + 0.22(6) = 2.17$;\enspace $\hat{y} = 10^{2.17} \approx 148$ million}

\end{notesbox}

\vspace{0.1in}

% ── STEP 3 ───────────────────────────────────────────────────────────────────
\begin{notesbox}{Step 3 --- The Power Model: Transform Both $x$ and $y$ (DAT-1.H)}
\small

\noindent \textbf{When to use:} The scatterplot curves like a power function ($y$ grows as a
power of $x$, e.g.\ area vs.\ radius).

\vspace{6pt}
\textbf{Transformation:} Apply $\log_{10}$ to \ans{both $x$ and $y$} values.

\vspace{4pt}
\textbf{Why it works:}
\[
  y = a\,x^b
  \;\;\Longrightarrow\;\;
  \log y = \log a + b\,\log x
\]

\vspace{6pt}
\textbf{Regression of $\log y$ on $\log x$} gives: \quad $\widehat{\log y} = $ \ans{$a + b(\log x)$}

\vspace{6pt}
\textbf{Back-transform:} \quad $\hat{y} = 10^{\widehat{\log y}}$

\vspace{6pt}
\renewcommand{\arraystretch}{1.6}
\begin{tabularx}{\linewidth}{@{} >{\bfseries}p{3cm} X X @{}}
\rowcolor{sky} \textbf{Model} & \textbf{Transformation} & \textbf{Regression} \\
Exponential & $\log y$ only & $\log y$ on $x$ \\
Power       & $\log y$ and $\log x$ & $\log y$ on $\log x$ \\
\end{tabularx}

\end{notesbox}

\vspace{0.1in}

% ── STEP 4 ───────────────────────────────────────────────────────────────────
\begin{notesbox}{Step 4 --- Checking the Transformation (DAT-1.G)}
\small

\noindent After transforming, \textbf{always} inspect the \ans{residual plot} of the
transformed model.

\vspace{4pt}
\begin{itemize}[itemsep=3pt]
  \item \textbf{Success:} residual plot shows \ans{random scatter (no pattern)} --- the transformation worked.
  \item \textbf{Failure:} residual plot still shows a \ans{curved pattern} --- try a different transformation.
\end{itemize}

\vspace{6pt}
Also compare \textbf{$r^2$}: a good transformation should \ans{increase} $r^2$ compared to
the original linear fit.

\end{notesbox}

\columnbreak

\textbf{Reflection:}\\
A friend says: ``I transformed $y$ and the slope of the transformed regression is $b = 0.22$.
That means $y$ increases by 0.22 for each unit increase in $x$.'' What is wrong with this
interpretation?\\[2pt]
\ansline{The slope 0.22 applies to $\log y$, not $y$ itself. On the original scale, the
relationship is multiplicative (exponential), not additive.}
