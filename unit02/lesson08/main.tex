\begin{tcolorbox}[colback=sky,colframe=navy,boxrule=0.9pt,arc=2mm,
    left=2mm,right=2mm,top=1.5mm,bottom=1.5mm]
    \textbf{Primary Objective:} Students will understand and apply the least-squares criterion,
    derive the slope and intercept formulas, and compute and interpret $r^2$ (the coefficient of
    determination) in context. Students will also evaluate the effect of influential points on the
    regression line (Big Idea: DAT; AP Skills 2 \& 4).
\end{tcolorbox}

\begin{skillbox}[Priority Ideas \& Skills]{goldbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP Skills 2 \& 4 --- Data Analysis \& Statistical Argumentation}
    \begin{itemize}
        \item Compute slope $b = r\dfrac{s_y}{s_x}$ and intercept $a = \bar{y} - b\bar{x}$.
        \item Interpret $r^2$: ``[value]\% of the variation in [y] is explained by the linear relationship with [x].''
        \item Identify and describe the effect of influential points on $r$, slope, and $r^2$.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Key Understandings}
    \begin{itemize}
        \item ``Least squares'' means the line minimizes $\sum(y_i - \hat{y}_i)^2$ --- the sum of squared residuals.
        \item The slope formula $b = r\frac{s_y}{s_x}$ connects regression directly to correlation.
        \item $r^2$ is the single most important model summary statistic --- it quantifies how well the model fits.
        \item Influential points can dramatically change the regression line; they must always be investigated, not just deleted.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[{Vocabulary, Concepts, \& Theorems}]{greenbox}
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Least-squares criterion}: choose the line that minimizes $\sum e_i^2 = \sum(y_i - \hat{y}_i)^2$
        \item \textbf{Slope formula}: $b = r\dfrac{s_y}{s_x}$
        \item \textbf{Intercept formula}: $a = \bar{y} - b\bar{x}$
        \item \textbf{Coefficient of determination} $r^2$: proportion of total variation in $y$ explained by the linear model; $0 \leq r^2 \leq 1$
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Influential point}: a point whose removal would substantially change the regression line (typically a point with extreme $x$ that is not consistent with the overall trend)
        \item \textbf{High leverage point}: a point with an extreme $x$ value (far from $\bar{x}$); may or may not be influential
        \item \textbf{$r^2 = 1$}: perfect fit; all variation in $y$ explained; \textbf{$r^2 = 0$}: no linear relationship; model no better than $\hat{y} = \bar{y}$
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Activate Prior Knowledge \& Spiral Review]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Quick Review (3 min)}\\
    Students recall from Lesson 2.7: the sum of residuals is 0. Ask: ``Why would we want to minimize $\sum e_i^2$ rather than $\sum e_i$?'' (The sum of residuals is always 0 regardless of the line --- squaring prevents cancellation of positive and negative errors.)

    \vspace{6pt}
    \textbf{Spiral Connections}
    \begin{itemize}
        \item Lesson 2.5: $r$ appears directly in the slope formula $b = r\frac{s_y}{s_x}$.
        \item Lesson 2.7: residuals are what we are minimizing.
        \item Unit 9: least-squares regression is the foundation for inference on the slope ($b$).
    \end{itemize}
\end{minipage}

\begin{skillbox}[Hook \& Lesson]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Hook (5 min)}\\
    Use an interactive tool (Desmos or class demonstration): place a movable line on a scatterplot and watch the sum of squared residuals displayed. Challenge students to find the line that minimizes the sum. Reveal: there is an exact formula --- the least-squares line.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Lesson Flow (18 min)}
    \begin{enumerate}
        \item State the least-squares criterion; explain why squaring is necessary.
        \item Derive the slope formula from summary statistics: $b = r\dfrac{s_y}{s_x}$.
        \item Compute intercept: $a = \bar{y} - b\bar{x}$.
        \item Compute $r^2$; interpret fully in context.
        \item Show how one influential point changes the slope, intercept, and $r^2$.
    \end{enumerate}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Explicit Instruction \& Active Monitoring]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Direct Instruction (10 min)}
    \begin{itemize}
        \item $r^2$ AP interpretation: ``About [value]\% of the variation in [response variable] is explained by the least-squares regression of [response] on [explanatory variable].''
        \item Influential points demo: show a scatterplot, compute $r^2$; add one far-right outlier; recompute. Discuss the dramatic change.
        \item Note: influential points are not automatically ``errors'' --- investigate context before removing.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Active Monitoring}
    \begin{itemize}
        \item Most common AP error: writing $r^2 = 0.85$ means ``the data are 85\% linear.'' Correct: 85\% of the \textit{variation} in $y$ is explained.
        \item Check formula application: Is $s_y$ in the numerator? Is $s_x$ in the denominator? Are $\bar{x}$ and $\bar{y}$ used in the intercept?
        \item Probe: ``If $r = 0.9$, what is $r^2$? Does a high $r^2$ guarantee the model is appropriate?'' (No --- check the residual plot.)
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Group Work \& Differentiation]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Group Activity (8--10 min)}\\
    Groups receive summary statistics ($\bar{x}$, $\bar{y}$, $s_x$, $s_y$, $r$). Groups:
    \begin{enumerate}
        \item Compute slope $b$ and intercept $a$.
        \item Write the regression equation with variable names.
        \item Compute and interpret $r^2$.
        \item Describe what it would mean for the model if an influential point were added at the far right.
    \end{enumerate}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Differentiation}
    \begin{itemize}
        \item \textit{Tier R}: Provide a reference card with the formulas and a step-by-step example; students follow the steps with new values.
        \item \textit{Tier A}: Students compute slope and intercept from summary statistics and interpret $r^2$ independently.
        \item \textit{Tier E}: Show two regression outputs with $r^2 = 0.87$ but very different residual plots. Students argue: which model is better, and why does $r^2$ alone not suffice?
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Individual Work \& Assessment]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Exit Ticket (5 min)}\\
    Given: $\bar{x} = 12$, $\bar{y} = 84$, $s_x = 3$, $s_y = 9$, $r = 0.80$. Students: (1) compute slope and intercept, (2) write the regression equation, (3) interpret $r^2$, (4) make a prediction for $x = 15$.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP-Style Check}\\
    \textit{``Computer output shows that $r^2 = 0.71$ for a regression of fuel efficiency on car weight. Interpret this value in context. Does a high $r^2$ mean the linear model is appropriate? Explain.''}\\[4pt]
    Full credit: correct $r^2$ interpretation with context; ``no'' with reference to checking the residual plot for patterns.
\end{minipage}
\end{skillbox}
