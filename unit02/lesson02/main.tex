    \textbf{Primary Objective:} Students will construct and interpret two-way (contingency)
    tables and segmented bar charts for two categorical variables, compute joint and
    conditional relative frequencies, and use these representations to determine whether
    the two variables are associated (Big Idea: UNC).

\renewcommand{\arraystretch}{1.6}
\begin{skillbox}[Priority Ideas \& Skills]{goldbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP Skill 2 --- Data Analysis}
    \begin{itemize}
        \item Construct a two-way table from raw data; identify joint and marginal
              frequencies (UNC-1.P.3).
        \item Compute joint and conditional relative frequencies from a contingency
              table (UNC-1.P.4, Skill 2.D).
        \item Compare conditional distributions using a segmented bar chart to
              determine whether an association exists (UNC-1.P.1, UNC-1.P.2).
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Key Understandings}
    \begin{itemize}
        \item A two-way table organizes two categorical variables simultaneously;
              row and column totals are marginal frequencies.
        \item A \emph{joint} relative frequency divides a cell count by the grand
              total; a \emph{conditional} relative frequency divides by a row or
              column total.
        \item Condition on the explanatory variable: if conditional distributions
              differ across groups, the variables are associated.
        \item A segmented bar chart built from conditional relative frequencies
              makes group comparisons visual and immediate.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[{Vocabulary, Concepts, \& Theorems}]{greenbox}
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Two-way table (contingency table)} --- organizes counts for two
              categorical variables; rows are one variable, columns are the other
        \item \textbf{Joint frequency} --- count in a single interior cell; describes
              both variables simultaneously
        \item \textbf{Marginal frequency} --- row or column total; describes one
              variable ignoring the other
        \item \textbf{Joint relative frequency} --- cell count divided by the grand
              total (UNC-1.P.4)
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Conditional relative frequency} --- cell count divided by the
              row or column total; holds one variable fixed
        \item \textbf{Segmented bar chart} --- each bar represents one group; bar
              divided into segments proportional to conditional relative frequencies
        \item \textbf{Association (categorical)} --- two variables are associated when
              conditional distributions differ across groups
        \item \textbf{Side-by-side bar chart / mosaic plot} --- alternative graphical
              displays for two categorical variables (UNC-1.P.1)
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Hook \& Lesson]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Hook (5 min)}\\
    Display a single question: \textit{``Are underclassmen and upperclassmen equally likely
    to eat in the school cafeteria?''}\\[4pt]
    Show 5 raw data rows (Grade, Lunch). Ask: ``How would you use this list to answer the
    question?'' Students immediately see a raw list is inefficient --- we need a
    two-way table to organize the counts.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Lesson Flow (18--20 min)}
    \begin{enumerate}
        \item Build the two-way table together from a sample of 240 students;
              label rows (grade: Under/Upper), columns (lunch: Cafeteria/Off-campus/Home).
        \item Identify joint frequencies (interior cells) and marginal frequencies (totals).
        \item Compute joint relative frequencies (each cell $\div$ 240).
        \item Compute conditional relative frequencies for each grade row
              (each cell $\div$ row total); compare rows --- do they differ?
        \item Construct a segmented bar chart from the conditional frequencies; interpret:
              if segments look different, the variables are associated.
    \end{enumerate}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Explicit Instruction \& Active Monitoring]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Direct Instruction (10 min)}
    \begin{itemize}
        \item Emphasize EK UNC-1.P.4: joint relative frequency $= \text{cell} / \text{grand total}$,
              \emph{not} the row or column total.
        \item Conditional relative frequency: divide by the row total when the row
              variable is explanatory (grade level explains lunch preference, not vice versa).
        \item Segmented bar construction: each bar $= 100\%$; segments colored by response
              variable (lunch type); bars labeled by explanatory variable (grade level).
        \item Association check: if the two bars' segment proportions look the same $\to$
              no evidence of association; if different $\to$ association.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Active Monitoring}
    \begin{itemize}
        \item Common error: dividing by the grand total to get conditional relative
              frequencies (gives joint, not conditional).
        \item Probe: \textit{``You divided by 240. What does that give you? (Joint relative freq.)
              To compare grade levels, what should you divide by?''} (row total = 120).
        \item Segmented bar check: do both bars reach 100\%? Are bars on the same scale?
        \item Cold-call: \textit{``Look at the two bars. Do they look the same or different?
              What does that tell you about association?''}
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Group Work \& Differentiation]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Group Activity (10--12 min)}\\
    Three differentiated tiers (activity sheet):
    \begin{itemize}
        \item \textit{Tier R}: Given a completed 2$\times$2 table (after-school job $\times$
              sports), compute conditional relative frequencies and answer: associated?
        \item \textit{Tier A}: Given raw counts, build a 2$\times$3 table (grade $\times$
              genre preference), compute conditional frequencies, sketch a segmented bar.
        \item \textit{Tier E}: Interpret a mosaic plot; compute marginal and conditional
              relative frequencies; write a two-sentence association claim.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Differentiation}
    \begin{itemize}
        \item \textit{Support}: Provide a computation scaffold with the formulas written
              in the table headers; totals column pre-filled.
        \item \textit{Extension}: Swap rows and columns in the same table; discuss how the
              association conclusion changes (or doesn't) when you condition on the other variable.
        \item \textit{Technology}: Use a spreadsheet to generate a segmented bar chart and
              compare its appearance to the hand-drawn version.
        \item \textit{ELL}: Pair vocabulary card (joint / marginal / conditional) with the table.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Individual Work \& Assessment]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Exit Ticket (5 min)}\\
    Given a partial two-way table (pet ownership $\times$ volunteering, $n = 180$):
    \begin{enumerate}
        \item Fill in the missing marginal totals.
        \item Compute the conditional relative frequencies for each pet-ownership group.
        \item State whether there is evidence of an association between pet ownership and
              volunteering; justify with specific values.
    \end{enumerate}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP-Style Check}\\
    \textit{``A two-way table shows grade level (9th/10th) and whether students participate
    in extracurriculars (Yes/No). For 9th graders, 60\% participate; for 10th graders, 55\%
    participate. Which of the following best describes the relationship?''}
    \begin{enumerate}[label=(\Alph*)]
        \item There is no association because both values are near 50\%.
        \item There is a strong association because the percentages are not equal.
        \item There is weak or no association because the conditional distributions are similar.
        \item Association cannot be determined without the actual counts.
    \end{enumerate}
    \textit{Answer: (C). Use student responses to calibrate readiness for Lesson 2.3.}
\end{minipage}
\end{skillbox}
