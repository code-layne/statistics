\small\textit{A health survey classified 330 adults by smoking status (Smoker / Non-smoker)
and whether they developed a respiratory condition (Yes / No) within five years.
Of 150 smokers, 45 developed the condition. Of 180 non-smokers, 18 developed the condition.}
\end{tcolorbox}

\vspace{0.2in}

\begin{enumerate}[leftmargin=1.6em, itemsep=0.2in]

    \item Compute the conditional proportion of adults who developed the condition for each group.\\[10pt]
          $\hat{p}_{\text{smoker}} = \dfrac{\blank{1.2cm}}{\blank{1.2cm}} = \blank{2.2cm}$
          \hfill
          $\hat{p}_{\text{non-smoker}} = \dfrac{\blank{1.2cm}}{\blank{1.2cm}} = \blank{2.2cm}$

    \item Compute the \textbf{difference in proportions} (smokers $-$ non-smokers) and interpret
          in context.\\[10pt]
          $\hat{p}_{\text{smoker}} - \hat{p}_{\text{non-smoker}} = \blank{2cm} - \blank{2cm} = \blank{2.4cm}$\\[8pt]
          Interpretation:\\[6pt]
          \writeline\\[2pt]\writeline

    \item Compute the \textbf{relative risk} of developing the condition for smokers compared
          to non-smokers and interpret in context.\\[10pt]
          $\text{RR} = \dfrac{\hat{p}_{\text{smoker}}}{\hat{p}_{\text{non-smoker}}}
          = \dfrac{\blank{2cm}}{\blank{2cm}} = \blank{2.4cm}$\\[8pt]
          Interpretation:\\[6pt]
          \writeline\\[2pt]\writeline

    \item Which statistic --- the difference in proportions or the relative risk --- better
          conveys the practical importance of the association here? Justify your answer in
          1--2 sentences.\\[8pt]
          \writeline\\[2pt]\writeline\\[2pt]\writeline

\end{enumerate}
