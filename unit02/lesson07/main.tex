    \textbf{Primary Objective:} Students will calculate and interpret residuals, construct and
    analyze residual plots, and use residual plots to assess whether a linear model is an
    appropriate model for a dataset (Big Idea: DAT; AP Skills 2.A \& 2.B).
\end{tcolorbox}

\begin{skillbox}[Priority Ideas \& Skills]{goldbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP Skill 2.B --- Data Analysis: Construct representations}
    \begin{itemize}
        \item Calculate a residual: $e = y - \hat{y}$.
        \item Interpret a residual in context: positive $\to$ actual is above the prediction
              (underpredicted); negative $\to$ actual is below the prediction (overpredicted).
        \item Construct a residual plot by plotting $(x,\, e)$ for each data point.
    \end{itemize}
    \textbf{AP Skill 2.A --- Data Analysis: Describe data}
    \begin{itemize}
        \item Describe the pattern in a residual plot (random scatter, curved, or fanning).
        \item Use the pattern to conclude whether a linear model is or is not appropriate.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Key Understandings (from DAT-1.E \& DAT-1.F)}
    \begin{itemize}
        \item The residual measures how far an observed value falls from the regression line:
              $e = y - \hat{y}$.
        \item The sum of all residuals for a least-squares line is always 0 --- a quick
              computational check.
        \item A residual plot with no pattern (random scatter around $e = 0$) is evidence
              that a linear form is appropriate (DAT-1.F.1).
        \item A curved or fanning pattern in a residual plot signals that a linear model is
              \textit{not} the best choice; another model should be considered (DAT-1.F.2).
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[{Vocabulary, Concepts, \& Theorems}]{greenbox}
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Residual}: $e = y - \hat{y}$; vertical distance from data point to
              regression line.
        \item \textbf{Positive residual}: actual $>$ predicted (point is above the line).
        \item \textbf{Negative residual}: actual $<$ predicted (point is below the line).
        \item \textbf{$\sum e_i = 0$}: residuals always sum to zero for the least-squares line.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Residual plot}: a scatterplot of residuals ($e$) vs.\ $x$ (or $\hat{y}$);
              used to assess model fit.
        \item \textbf{Random scatter}: residual plot showing no pattern $\to$ linear model
              is appropriate.
        \item \textbf{Curved pattern}: residual plot showing a curve $\to$ a nonlinear model
              may provide a better fit.
        \item \textbf{Fanning (heteroscedasticity)}: spread of residuals grows with $x$ $\to$
              equal-variance assumption is violated.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Hook]{sky}
\begin{minipage}[t]{0.56\linewidth}
    \textbf{Hook (3--5 min)}\\
    Display a scatterplot (sunlight hours vs.\ temperature) with the regression line drawn.
    Point to a data point \textit{below} the line and ask:
    \begin{itemize}
        \item ``Is the actual temperature higher or lower than our prediction?''
        \item ``By how much?''
    \end{itemize}
    Students see that the \textit{vertical gap} between the point and the line is meaningful
    --- it measures prediction error. That gap is a residual.
\end{minipage}
\hfill
\begin{minipage}[t]{0.40\linewidth}
    \textbf{Entry question:}\\
    The regression line predicts 66°F for a city with 10 hours of sunlight. The actual
    average high is 66°F. What is the residual? What does it mean?
\end{minipage}
\end{skillbox}

\begin{skillbox}[Lesson]{sky}
\begin{multicols}{2}
\textbf{1.\ Define the residual (5 min)}

Present: $e = y - \hat{y}$. Ask:
\begin{itemize}
    \item ``If the actual is \textit{above} the line, is $e$ positive or negative?''
    \item ``If the actual is \textit{below} the line?''
\end{itemize}
Model one calculation using the sunlight/temperature dataset.

\textbf{2.\ Compute all residuals (8 min)}

Work through the full table together --- record $x$, $y$, $\hat{y}$, and $e$ for all 5
cities. Ask students to verify $\sum e = 0$ as a group check.

\textbf{3.\ Construct the residual plot (7 min)}

Plot $(x,\, e)$ on new axes (label: $x$-axis = ``daily sunlight (hr)'',
$y$-axis = ``residual (°F)''). Draw a dashed horizontal line at $e = 0$.

\textbf{4.\ Interpret the residual plot (5 min)}

Pose four residual plots (random, curved upward, fanning, curved downward).
For each: name the pattern, then conclude.
\begin{itemize}
    \item \textbf{Random scatter} $\to$ ``A linear model appears appropriate.''
    \item \textbf{Curved pattern} $\to$ ``The residual plot shows a curved pattern, suggesting
          a nonlinear model may provide a better fit.''
    \item \textbf{Fanning} $\to$ ``The spread of residuals increases with $x$, which may
          indicate a linear model is not appropriate.''
\end{itemize}
\end{multicols}
\end{skillbox}

\begin{skillbox}[Explicit Instruction: AP Response Template]{sky}
\begin{minipage}[t]{0.50\linewidth}
    \textbf{Residual plot interpretation rubric (3 elements)}
    \begin{enumerate}
        \item \textbf{Pattern}: describe what you see (random scatter / curved / fanning).
        \item \textbf{Conclusion}: state whether a linear model is or is not appropriate.
        \item \textbf{Context}: mention the variables by name.
    \end{enumerate}
    \medskip
    Model full response (curved pattern):
    \begin{tcolorbox}[colback=white,colframe=navy,boxrule=0.6pt,arc=1.5mm,
        left=2mm,right=2mm,top=1mm,bottom=1mm]
        \small``The residual plot shows a curved (U-shaped) pattern. This suggests that a
        linear model may \textit{not} be an appropriate model for predicting [response]
        from [explanatory]. A nonlinear model may provide a better fit.''
    \end{tcolorbox}
\end{minipage}
\hfill
\begin{minipage}[t]{0.46\linewidth}
    \textbf{Random-scatter response:}
    \begin{tcolorbox}[colback=white,colframe=navy,boxrule=0.6pt,arc=1.5mm,
        left=2mm,right=2mm,top=1mm,bottom=1mm]
        \small``The residual plot shows no apparent pattern (random scatter around zero).
        This suggests that a linear model \textit{is} an appropriate model for predicting
        [response] from [explanatory].''
    \end{tcolorbox}
    \medskip
    \textbf{AP trap:} Describing the direction of the curve (``opens upward'') without
    concluding about the model earns partial credit only. Always link the pattern to
    the model assessment.
\end{minipage}
\end{skillbox}

\begin{skillbox}[Active Monitoring]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Circulate and check:}
    \begin{itemize}
        \item Are students subtracting $y - \hat{y}$ (not $\hat{y} - y$)?
        \item Do residuals sum to 0? (If not, a $\hat{y}$ was computed incorrectly.)
        \item Is the residual plot labeled? ($x$-axis = explanatory variable or $\hat{y}$;
              $y$-axis = residual.)
        \item Does the response include all three AP elements: pattern, conclusion, context?
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Cold-call prompts:}
    \begin{itemize}
        \item ``A residual of $-3$ means the actual value is\ldots?''
        \item ``The residuals sum to $+2$. What likely went wrong?''
        \item ``The residual plot shows a clear U-shape. What does that tell you?''
        \item ``Why do we plot residuals vs.\ $x$ (not $y$ vs.\ $x$ again)?''
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Group Work \& Differentiation]{redbox}
\begin{multicols}{3}
\textbf{Tier R --- Remediate}
\begin{itemize}
    \item Receive a table with $\hat{y}$ pre-filled; compute $e = y - \hat{y}$ only.
    \item Verify $\sum e = 0$.
    \item Describe a pre-drawn residual plot from the pattern options (random / curved /
          fanning).
\end{itemize}

\columnbreak

\textbf{Tier A --- Approaching}
\begin{itemize}
    \item Given a dataset and equation, compute $\hat{y}$ and $e$ for all points.
    \item Construct the residual plot on a provided grid.
    \item Write a complete AP-style conclusion about model appropriateness.
\end{itemize}

\columnbreak

\textbf{Tier E --- Extension}
\begin{itemize}
    \item Compare two datasets: one with random residuals, one with a curved pattern.
    \item Argue which linear model is more appropriate and justify using the residual plots.
    \item Identify the point with the largest absolute residual and interpret it in context.
\end{itemize}
\end{multicols}
\end{skillbox}

\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP-Style MC:}\\
    \small\textit{``A residual plot for a regression of $y$ = daily step count on $x$ = hours
    of physical activity shows a clear curved pattern. What does this indicate?''}
    \begin{enumerate}[label=(\Alph*)]
        \item The least-squares line has a positive slope.
        \item A linear model is an appropriate model for the data.
        \item A linear model may not be an appropriate model for the data.
        \item The residuals do not sum to zero.
    \end{enumerate}
    \textbf{Answer:} (C). Reason: a curved residual plot indicates the linear model is not
    capturing all the structure in the data.
\end{minipage}
