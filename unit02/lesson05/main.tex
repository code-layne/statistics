\begin{tcolorbox}[colback=sky,colframe=navy,boxrule=0.9pt,arc=2mm,
    left=2mm,right=2mm,top=1.5mm,bottom=1.5mm]
    \textbf{Primary Objective:} Students will calculate and interpret the correlation
    coefficient $r$ for a linear relationship between two quantitative variables ---
    understanding that $r$ quantifies the strength and direction of a \emph{linear}
    association, is unit-free, and does not imply causation (Big Idea: DAT; AP Skills 2.C \& 4.B).
\end{tcolorbox}

\begin{skillbox}[Priority Ideas \& Skills]{goldbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP Skills 2.C \& 4.B --- Data Analysis \& Statistical Argumentation}
    \begin{itemize}
        \item Calculate $r$ using technology (or the formula) for a bivariate dataset.
        \item Interpret $r$: state direction (sign), strength (magnitude), and that the
              association is linear --- in context.
        \item Explain why a value of $r$ near $\pm 1$ does \emph{not} guarantee that a
              linear model is appropriate.
        \item Distinguish association from causation.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Key Understandings}
    \begin{itemize}
        \item $r$ is always between $-1$ and $1$, inclusive, and is unit-free.
        \item The sign of $r$ gives direction; the magnitude gives strength of the
              \emph{linear} association.
        \item $r = 0$ indicates no \emph{linear} association (curved patterns can still exist).
        \item A perceived relationship between two variables does not mean one \emph{causes}
              the other (DAT-1.C.2).
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[{Vocabulary, Concepts, \& Theorems}]{greenbox}
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{Correlation coefficient ($r$)} --- a number between $-1$ and $1$ that
              measures the direction and strength of the linear association between two
              quantitative variables
        \item \textbf{Positive correlation} ($r > 0$) --- as $x$ increases, $y$ tends to increase
        \item \textbf{Negative correlation} ($r < 0$) --- as $x$ increases, $y$ tends to decrease
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \begin{itemize}
        \item \textbf{No linear association} ($r \approx 0$) --- knowing $x$ does not
              linearly predict $y$
        \item \textbf{Perfect linear association} ($r = \pm 1$) --- all points lie exactly
              on a line
        \item \textbf{Correlation $\neq$ causation} --- a strong $r$ does not mean one
              variable causes changes in the other
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Activate Prior Knowledge \& Spiral Review]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Quick Review (3 min)}\\
    Students describe the direction and strength of three scatterplots (from Lesson 2.4
    DOFS language). Ask: ``Wouldn't it be useful to have a single number to compare
    how strong each relationship is?'' --- motivates $r$.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Spiral Connections}
    \begin{itemize}
        \item Lesson 2.4: direction and strength from DOFS --- $r$ puts a number on those words.
        \item Lessons 2.6--2.7: the regression line relies on $r$ (slope $= r \cdot s_y/s_x$).
        \item Unit 3: inference for regression uses $r^2$ and the $t$-test for slope.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Hook \& Lesson]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Hook (5 min)}\\
    Display four scatterplots on the board. Ask students to rank them from ``strongest
    relationship'' to ``weakest.'' After discussion, reveal $r$ values and confirm (or
    surprise!) the class. Debrief: $r$ makes ``strong'' and ``weak'' precise.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Lesson Flow (15 min)}
    \begin{enumerate}
        \item Introduce the formula; emphasize technology is the usual tool (EK DAT-1.B.2).
        \item Students calculate $r$ for a small dataset by hand once to build intuition.
        \item Display several $(r, \text{scatterplot})$ pairs; students identify matches.
        \item Discuss: can $r$ be near $\pm 1$ for a curved dataset? (EK DAT-1.B.3)
        \item Address causation with a classic lurking-variable example.
    \end{enumerate}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Explicit Instruction \& Active Monitoring]{sky}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Direct Instruction (10 min)}
    \begin{itemize}
        \item Write the formula:
              $r = \dfrac{1}{n-1}\sum\!\left(\dfrac{x_i - \bar x}{s_x}\right)\!\left(\dfrac{y_i - \bar y}{s_y}\right)$
        \item Emphasize: $r$ is unit-free (standardized $z$-scores); swapping $x$ and $y$
              leaves $r$ unchanged.
        \item Model a full interpretation: ``There is a [strong/moderate/weak]
              [positive/negative] linear association between [var1] and [var2]; $r = [\ldots]$.''
        \item Show a dataset with $r \approx 0.9$ but a curved pattern to reinforce
              EK DAT-1.B.3.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Active Monitoring}
    \begin{itemize}
        \item Common errors: forgetting to state ``linear'' in the interpretation; confusing
              $r$ with $r^2$; concluding causation from a high $r$.
        \item Prompt: ``Does $r$ tell us the \emph{form} of the relationship? What if
              the scatterplot is curved?''
        \item Circulate as students calculate $r$ for the activity dataset; look for
              sign errors.
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Group Work \& Differentiation]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Group Activity (10 min)}\\
    Each tier receives a different dataset. Groups:
    \begin{enumerate}
        \item Compute $r$ using technology (and verify with formula for Tier E).
        \item Interpret $r$ in context: direction, strength, linear qualifier.
        \item Sketch a scatterplot that could match their $r$; compare with the actual plot.
        \item Respond to: ``Does this $r$ prove one variable causes the other?''
    \end{enumerate}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Differentiation}
    \begin{itemize}
        \item \textit{Support (Tier R)}: pre-computed $\bar x$, $\bar y$, $s_x$, $s_y$ provided;
              students complete the formula with guided steps.
        \item \textit{Extension (Tier E)}: students generate their own dataset targeting
              a specified $r$ value, then verify with technology.
        \item \textit{Misconception check}: provide two scenarios with the same $r$ but
              different scatterplots (one linear, one curved) --- which model is appropriate?
    \end{itemize}
\end{minipage}
\end{skillbox}

\begin{skillbox}[Individual Work \& Assessment]{redbox}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Exit Ticket (5 min)}\\
    Students are given $r = -0.87$ for a dataset of engine size ($x$, liters) vs.\ fuel
    efficiency ($y$, mpg) for 30 cars. They write a complete interpretation and answer
    whether this proves larger engines \emph{cause} lower efficiency.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{AP-Style Check}\\
    \textit{``Describe the linear association between engine size and fuel efficiency using
    the correlation coefficient.''}\\[4pt]
    Model: ``There is a strong negative linear association between engine size and fuel
    efficiency for these 30 cars ($r = -0.87$). As engine size increases, fuel efficiency
    tends to decrease. This does not prove that larger engines \emph{cause} lower efficiency ---
    other factors may be involved.''
\end{minipage}
\end{skillbox}

        \item Interpret given $r$ values in 3 contexts; identify direction, strength, and
              whether causation can be concluded.
        \item Match five scatterplot sketches to five $r$ values.
        \item Investigate a real dataset (provided): compute $r$ with technology and write
              a full interpretation.
    \end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Extension / Preview}
    \begin{itemize}

              films vs.\ drowning deaths). Write a paragraph explaining why the high $r$ is
              misleading.
        \item \textit{Preview}: Lesson 2.6 uses $r$ to build the \textbf{least-squares
              regression line} --- a formula that makes predictions from $x$ values.
    \end{itemize}
\end{minipage}
\end{skillbox}
