\begin{vocabbox}
\begin{description}
  \item[\termblanklong{Categorical inference procedure}]
        Any inference method applied to \blank{4cm} data. The first diagnostic question is:
        does the outcome have \blank{2cm} categories or \blank{3cm}?

  \item[\termblanklong{Chi-square goodness-of-fit (GOF) test}]
        Used when there is \blank{2cm} sample and \blank{2cm} or more categories. Tests
        whether the observed distribution matches a \blank{5cm} distribution.

  \item[\termblanklong{Chi-square test for homogeneity}]
        Used when there are \blank{2.5cm} or more \blank{3cm} samples and one categorical
        variable. Tests whether the \blank{4.5cm} is the same across groups.

  \item[\termblanklong{Chi-square test for independence}]
        Used when there is \blank{2cm} sample and \blank{2cm} categorical variables recorded
        for each subject. Tests whether there is an \blank{4cm} between the variables.
\end{description}
\end{vocabbox}
