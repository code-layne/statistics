\begin{enumerate}

  \item Heart rates by age group.
        \begin{enumerate}[label=\alph*.]
          \item \ans{Random: both stated. 10\%: both $\le 10\%$ of respective age groups. Independent groups. Normal/Large: both populations approximately normal; also $n_1=55\ge30$ and $n_2=45\ge30$. All conditions met.}
          \item \ans{$\mu = 70-74 = -4$ bpm; $\sigma = \sqrt{121/55+196/45} = \sqrt{2.2+4.356} \approx \sqrt{6.556} \approx 2.560$ bpm.}
          \item \ans{$z = (-8-(-4))/2.560 \approx -1.563$; $P < -8 \approx 0.0591 \approx 5.9\%$.}
          \item \ansline{There is approximately a 5.9\% chance that the sample mean heart rate for adults 20--39 falls more than 8 bpm below the sample mean for adults 60--79, even if the true difference is only 4 bpm.}
        \end{enumerate}

  \item Light bulbs.
        \begin{enumerate}[label=\alph*.]
          \item \ans{$\mu = 1200-1050=150$ hr; $\sigma = \sqrt{14400/36+22500/49} = \sqrt{400+459.18} \approx \sqrt{859.18} \approx 29.31$ hr. $z=(200-150)/29.31\approx1.706$; $P>200\approx1-0.9560=0.0440\approx4.4\%$.}
          \item \ans{$z=\text{invNorm}(0.95)\approx1.645$; $c=150+1.645(29.31)\approx198.2$ hr. In 5\% of repeated sampling, the difference in sample mean lifetimes exceeds 198.2 hours.}
        \end{enumerate}

\end{enumerate}
