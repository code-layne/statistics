\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\textbf{Context: Comparing plant growth.}
Fertilizer A plants: $\mu_1 = 24$ cm, $\sigma_1 = 4$ cm, $n_1 = 25$ (pop.\ normal).
Fertilizer B plants: $\mu_2 = 20$ cm, $\sigma_2 = 5$ cm, $n_2 = 25$ (pop.\ normal).

\begin{enumerate}
  \item Check all conditions for using the normal model for $\bar{x}_1 - \bar{x}_2$.
        \writelines{3}
  \item Find $\mu_{\bar{x}_1-\bar{x}_2}$ and $\sigma_{\bar{x}_1-\bar{x}_2}$.
        \vspace{0.6in}
  \item Find $P(\bar{x}_1 - \bar{x}_2 > 6)$.
        \vspace{0.8in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\textbf{Context: Comparing study methods.}
Students using Method A (active recall): $\mu_1 = 82$, $\sigma_1 = 10$, $n_1 = 40$.
Students using Method B (re-reading): $\mu_2 = 76$, $\sigma_2 = 12$, $n_2 = 50$.
Both populations are approximately normal.

\begin{enumerate}
  \item Check all conditions.
        \writelines{3}
  \item Find $\mu$ and $\sigma$ for $\bar{x}_1 - \bar{x}_2$.
        \vspace{0.5in}
  \item Find $P(\bar{x}_1 - \bar{x}_2 > 10)$.
        \vspace{0.8in}
  \item Find $P(\bar{x}_1 - \bar{x}_2 < 0)$ and interpret. What does a negative difference
        mean in this context?
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\textbf{Context: City commute times.}
City 1 commuters (right-skewed): $\mu_1 = 35$ min, $\sigma_1 = 12$, $n_1 = 60$.
City 2 commuters (right-skewed): $\mu_2 = 28$ min, $\sigma_2 = 9$, $n_2 = 45$.

\begin{enumerate}
  \item Check all conditions, including whether CLT applies to each group.
        \writelines{3}
  \item Find $\mu$ and $\sigma$ for $\bar{x}_1 - \bar{x}_2$.
        \vspace{0.5in}
  \item Find $P(5 < \bar{x}_1 - \bar{x}_2 < 10)$.
        \vspace{0.8in}
  \item The 2.5th percentile of the sampling distribution of $\bar{x}_1 - \bar{x}_2$:
        find this value and interpret.
        \vspace{0.8in}
  \item If both $n_1$ and $n_2$ are doubled, describe what happens to $\sigma_{\bar{x}_1-\bar{x}_2}$.
        \vspace{0.6in}
\end{enumerate}
\end{tcolorbox}
