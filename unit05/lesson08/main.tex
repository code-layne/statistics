\begin{objectivebox}
\textbf{Topic:} 5.8 --- Sampling Distributions for Differences in Sample Means\\
\textbf{Enduring Understanding:} UNC-3\\
\textbf{Learning Objectives:} UNC-3.T, UNC-3.U, UNC-3.V\\
\textbf{Essential Knowledge:} UNC-3.T.1; UNC-3.U.1--U.2; UNC-3.V.1--V.3\\
\textbf{Mathematical Practice:} 3.C --- Verify conditions; 4.B --- Connect representations
\end{objectivebox}

\begin{learningtargetbox}
By the end of this lesson, students will be able to:
\begin{itemize}
  \item State $\mu_{\bar{x}_1-\bar{x}_2} = \mu_1 - \mu_2$ and
        $\sigma_{\bar{x}_1-\bar{x}_2} = \sqrt{\sigma_1^2/n_1 + \sigma_2^2/n_2}$.
  \item Check conditions (random, 10\%, Normal/Large Sample for each group).
  \item Compute probabilities for $\bar{x}_1 - \bar{x}_2$ using the normal model.
\end{itemize}
\end{learningtargetbox}

\begin{vocabbox}
sampling distribution of $\bar{x}_1 - \bar{x}_2$, independence between groups, Normal/Large Sample condition for each group
\end{vocabbox}

\begin{hookbox}
\textbf{Hook --- Gender Sleep Difference (5 min):} Women average 7.1 hours of sleep
($\mu_1 = 7.1$, $\sigma_1 = 1.2$) and men average 6.8 hours ($\mu_2 = 6.8$,
$\sigma_2 = 1.4$). If we randomly sample 50 women and 50 men, how often would the
sample difference $\bar{x}_1 - \bar{x}_2$ exceed 1 hour?
\end{hookbox}

\begin{notesbox}
\textbf{Direct Instruction + Activity (35 min):} State formulas parallel to the two-proportion
case. Emphasize independence between groups. Work full example. Tiered activity is the
culminating lesson for the unit.
\end{notesbox}

\end{reflectionbox}
