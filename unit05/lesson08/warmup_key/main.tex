  \item \ansline{$\mu_A - \mu_B = 540 - 500 = 40$. Since $\bar{x}_A$ is an unbiased estimator of $\mu_A$ and $\bar{x}_B$ is an unbiased estimator of $\mu_B$, the difference $\bar{x}_A - \bar{x}_B$ is an unbiased estimator of $\mu_A - \mu_B = 40$.}

\end{enumerate}
