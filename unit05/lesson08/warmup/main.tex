\begin{enumerate}

  \item Scores on a standardized test are normally distributed with $\mu = 500$ and
        $\sigma = 100$. A random sample of $n = 25$ students is selected. Find
        $\mu_{\bar{x}}$, $\sigma_{\bar{x}}$, and $P(\bar{x} > 520)$.
        \vspace{1.2in}

  \item Two groups of students are tested: Group A (tutored, $\mu_A = 540$) and Group B
        (not tutored, $\mu_B = 500$). If you take random samples of 25 from each group,
        what would you expect $\bar{x}_A - \bar{x}_B$ to equal on average? Explain.
        \writelines{2}

\end{enumerate}
