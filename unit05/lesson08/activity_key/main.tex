\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\begin{enumerate}
  \item \ans{Random: stated. 10\%: both $\le 10\%$ of all plants. Independent groups. Normal/Large: both populations normal. All conditions met.}
  \item \ans{$\mu = 24-20 = 4$ cm; $\sigma = \sqrt{16/25+25/25} = \sqrt{0.64+1.00} = \sqrt{1.64} \approx 1.281$ cm.}
  \item \ans{$z = (6-4)/1.281 \approx 1.562$; $P > 6 \approx 1-0.9409 = 0.0591 \approx 5.9\%$.}
\end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\begin{enumerate}
  \item \ans{Random: stated. 10\%: both $\le 10\%$ of student populations. Independent groups. Normal/Large: both populations approximately normal; also $n_1=40\ge30$ and $n_2=50\ge30$. All conditions met.}
  \item \ans{$\mu = 82-76 = 6$; $\sigma = \sqrt{100/40+144/50} = \sqrt{2.50+2.88} = \sqrt{5.38} \approx 2.320$.}
  \item \ans{$z = (10-6)/2.320 \approx 1.724$; $P > 10 \approx 1-0.9576 = 0.0424 \approx 4.2\%$.}
  \item \ans{$z = (0-6)/2.320 \approx -2.586$; $P < 0 \approx 0.0049 \approx 0.5\%$. A negative difference means the re-reading group outscored the active recall group --- very unlikely given the true difference, but possible due to sampling variability.}
\end{enumerate}
\end{tcolorbox}

\vspace{0.1in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\begin{enumerate}
  \item \ans{Random: stated. 10\%: both $\le 10\%$ of city commuters. Independent: separate cities. Normal/Large: $n_1=60\ge30$ and $n_2=45\ge30$; CLT applies to both. All conditions met.}
  \item \ans{$\mu = 35-28 = 7$ min; $\sigma = \sqrt{144/60+81/45} = \sqrt{2.40+1.80} = \sqrt{4.20} \approx 2.049$ min.}
  \item \ans{$z_1=(5-7)/2.049\approx-0.976$; $z_2=(10-7)/2.049\approx1.464$; $P\approx0.9285-0.1647=0.7638\approx76.4\%$.}
  \item \ans{$z=\text{invNorm}(0.025)\approx-1.960$; $c=7+(-1.960)(2.049)\approx2.98$ min. In 2.5\% of repeated sampling scenarios, the sample mean commute difference falls below 2.98 minutes.}
  \item \ans{Doubling both $n$'s multiplies each variance by $1/2$, so $\sigma_{\bar{x}_1-\bar{x}_2}$ is multiplied by $1/\sqrt{2} \approx 0.707$ --- it decreases by about 29\%.}
\end{enumerate}
\end{tcolorbox}
