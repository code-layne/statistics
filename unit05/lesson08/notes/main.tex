\begin{objectivebox}
\textbf{Learning Targets}
\begin{enumerate}
  \item I can state the center and spread of the sampling distribution of $\bar{x}_1 - \bar{x}_2$.
  \item I can check conditions for using the normal model.
  \item I can compute probabilities for $\bar{x}_1 - \bar{x}_2$.
\end{enumerate}
\end{objectivebox}

\begin{notesbox}
\textbf{Sampling Distribution of $\bar{x}_1 - \bar{x}_2$} (UNC-3.T, U, V)

\medskip
\begin{tabular}{@{} p{0.2\linewidth} p{0.72\linewidth} @{}}
\textbf{Center:} & $\mu_{\bar{x}_1-\bar{x}_2} = $ \blank{1.5in} \\[12pt]
\textbf{Spread:} & $\sigma_{\bar{x}_1-\bar{x}_2} = $ \TallMath{\sqrt{\dfrac{\blank{0.5in}^2}{\blank{0.4in}} + \dfrac{\blank{0.5in}^2}{\blank{0.4in}}}} \\[16pt]
\textbf{Shape:} & Normal/approximately normal when both groups satisfy the \blank{2.0in} condition. \\
\end{tabular}
\end{notesbox}

\begin{notesbox}
\textbf{Conditions for $\bar{x}_1 - \bar{x}_2$}

\begin{enumerate}
  \item \textbf{Random (both groups):} each sample is a random sample.
  \item \textbf{10\% condition (both groups):} $n_1 \le 0.10 N_1$ and $n_2 \le 0.10 N_2$.
  \item \textbf{Independence between groups:} the two samples are \blank{1.5in}.
  \item \textbf{Normal/Large Sample (each group):} each population is normal OR each $n \ge$ \blank{0.4in}.
\end{enumerate}
\end{notesbox}
