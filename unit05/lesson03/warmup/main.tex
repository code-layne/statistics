\vspace{0.15in}

\begin{enumerate}

  \item A population of fish has lengths that are approximately normally distributed with
        $\mu = 14$ cm and $\sigma = 3$ cm.
        \begin{enumerate}[label=\alph*.]
          \item What is the probability that a single randomly selected fish is longer than 17 cm?
                \vspace{1.0in}
          \item A researcher catches a random sample of 9 fish. Without any formula, predict:
                Would the \emph{average} length of 9 fish vary more or less than a single fish's
                length? Explain.
                \writelines{2}
        \end{enumerate}

  \item A dot plot shows sample means $\bar{x}$ from 200 random samples of size $n = 30$
        drawn from a right-skewed population. Predict the shape of this dot plot and explain
        your reasoning.
        \writelines{3}

\end{enumerate}
