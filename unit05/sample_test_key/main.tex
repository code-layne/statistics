\noindent\textbf{Part II --- Free Response}

\begin{enumerate}
\setcounter{enumi}{10}

  \item Exercise habits, $p = 0.28$, $n = 200$.

        \begin{enumerate}[label=(\alph*)]
          \item \ans{(1) \textbf{Random:} stated --- the sample is a random sample. \checkmark\\
                (2) \textbf{10\% condition:} $n = 200 \le 10\%$ of all adults (millions). \checkmark\\
                (3) \textbf{Large Counts:} $np = 200(0.28) = 56 \ge 10$; $n(1-p) = 200(0.72) = 144 \ge 10$. \checkmark\\
                All conditions met; use the normal model for $\hat{p}$.}

          \item \ans{$\mu_{\hat{p}} = 0.28$;\quad $\sigma_{\hat{p}} = \sqrt{0.28(0.72)/200} = \sqrt{0.001008} \approx 0.03175$.}

          \item \ans{$z = (0.32 - 0.28)/0.03175 \approx 1.260$. $P(\hat{p}>0.32) = P(Z>1.260) \approx 1 - 0.8962 = 0.1038$. About 10.4\%.}

          \item \ansline{If the true proportion of adults who exercise at least 5 days per week is 28\%, there is approximately a 10.4\% chance that a random sample of 200 adults would yield a sample proportion of more than 32\%.}
        \end{enumerate}

\begin{teachernote}
Award partial credit on (c) if the student correctly standardizes but makes a minor arithmetic error. Require a sentence with context for full credit on (d).
\end{teachernote}

  \item Bolt lengths $N(50, 1.5)$.

        \begin{enumerate}[label=(\alph*)]
          \item \ans{$z = (51.5-50)/1.5 = 1.00$; $P(X>51.5) = 1-0.8413 = 0.1587$. About 15.9\%.}

          \item \ans{(1) \textbf{Random:} stated. \checkmark\\
                (2) \textbf{10\%:} $36 \le 10\%$ of all bolts produced. \checkmark\\
                (3) \textbf{Normal/Large Sample:} population is approximately normal. \checkmark\\
                All conditions met.}

          \item \ans{$\mu_{\bar{x}} = 50$ mm;\quad $\sigma_{\bar{x}} = 1.5/\sqrt{36} = 0.25$ mm.}

          \item \ans{$z = (50.5-50)/0.25 = 2.00$; $P(\bar{x}>50.5) = 1-0.9772 = 0.0228$. About 2.3\%.}

          \item \ansline{The probability for a single bolt (15.9\%) is much larger than for the sample mean (2.3\%). Averaging 36 bolts cancels out individual extremes: the standard error ($\sigma_{\bar{x}} = 0.25$) is much smaller than the population SD ($\sigma = 1.5$), so sample means are far less variable than individual measurements.}

          \item \ans{$z = (49.4-50)/0.25 = -2.40$; $P(\bar{x}<49.4) = P(Z<-2.40) \approx 0.0082$. About 0.82\%. This is unusual (probability $< 5\%$); such a sample mean would occur by chance less than 1\% of the time and warrants investigation.}
        \end{enumerate}

  \item Homework completion, two schools.

        \begin{enumerate}[label=(\alph*)]
          \item \ans{$n_1 p_1 = 120(0.65) = 78 \ge 10$. \checkmark\\
                $n_1(1-p_1) = 120(0.35) = 42 \ge 10$. \checkmark\\
                $n_2 p_2 = 150(0.58) = 87 \ge 10$. \checkmark\\
                $n_2(1-p_2) = 150(0.42) = 63 \ge 10$. \checkmark\\
                All four Large Counts conditions met.}

          \item \ans{$\mu_{\hat{p}_1-\hat{p}_2} = 0.65 - 0.58 = 0.07$.\\
                $\sigma = \sqrt{0.65(0.35)/120 + 0.58(0.42)/150} = \sqrt{0.001896 + 0.001624} = \sqrt{0.003520} \approx 0.05933$.}

          \item \ans{$z = (0.15 - 0.07)/0.05933 \approx 1.348$; $P(\hat{p}_1-\hat{p}_2>0.15) \approx 1-0.9112 = 0.0888 \approx 8.9\%$.}
        \end{enumerate}

\end{enumerate}
