\begin{enumerate}

  \item Daily steps, $n = 64$.
        \begin{enumerate}[label=\alph*.]
          \item \ans{Random: stated. 10\%: $64 \le 10\%$ of all adults --- yes. Normal/Large: $n=64\ge30$ --- yes. Conditions met.}
          \item \ans{$\mu_{\bar{x}} = 7500$; $\sigma_{\bar{x}} = 2000/\sqrt{64} = 250$ steps.}
          \item \ans{$z = (7000-7500)/250 = -2.00$; $P < 7000 = 0.0228$. About 2.3\%.}
          \item \ans{$z = 1.282$; $c = 7500 + 1.282(250) \approx 7821$ steps. In 10\% of samples of 64 adults, the mean daily step count exceeds 7,821 steps.}
        \end{enumerate}

  \item Exam scores, $N(74,12)$, $n=9$.
        \begin{enumerate}[label=\alph*.]
          \item \ans{Random: stated. 10\%: $9 \le 10\%$ of all exams. Normal/Large: population is normal --- yes. Conditions met.}
          \item \ans{$\sigma_{\bar{x}} = 12/\sqrt{9} = 4$. $z_1 = (70-74)/4 = -1.00$; $z_2 = (78-74)/4 = 1.00$; $P = 0.8413 - 0.1587 = 0.6826 \approx 68.3\%$.}
        \end{enumerate}

  \item $\mu=200$, $\sigma=40$.
        \begin{enumerate}[label=\alph*.]
          \item \ans{$n=25$: $\sigma_{\bar{x}}=8$. $n=100$: $\sigma_{\bar{x}}=4$. $n=400$: $\sigma_{\bar{x}}=2$.}
          \item \ans{$z_1=(195-200)/4=-1.25$; $z_2=(205-200)/4=1.25$; $P=0.8944-0.1056=0.7888\approx78.9\%$.}
        \end{enumerate}

\end{enumerate}

\begin{extensionbox}
\textbf{Extension.}
\ans{Need $P(|\bar{x}-200|<5)\ge0.95$, i.e., $5/\sigma_{\bar{x}}\ge1.960$, so $\sigma_{\bar{x}}\le5/1.960\approx2.551$. Then $40/\sqrt{n}\le2.551\Rightarrow\sqrt{n}\ge15.68\Rightarrow n\ge246$.}
\end{extensionbox}
