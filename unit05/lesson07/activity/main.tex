\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\textbf{Context: Vitamin C content in oranges.}
Oranges in a large shipment have vitamin C content distributed approximately normally
with $\mu = 70$ mg and $\sigma = 8$ mg. A grocery store tests a random sample of
$n = 16$ oranges.

\begin{enumerate}
  \item Check all three conditions.
        \writelines{3}
  \item Find $\mu_{\bar{x}}$ and $\sigma_{\bar{x}}$.
        \vspace{0.5in}
  \item Find $P(\bar{x} > 73)$.
        \vspace{0.8in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\textbf{Context: Customer wait times.}
Wait times at a bank branch are right-skewed with $\mu = 8.2$ min and $\sigma = 3.5$ min.
A manager monitors a random sample of $n = 50$ customers on a given day.

\begin{enumerate}
  \item Check all conditions and justify whether the normal model is appropriate.
        \writelines{3}
  \item Find $\mu_{\bar{x}}$ and $\sigma_{\bar{x}}$.
        \vspace{0.5in}
  \item Find $P(\bar{x} > 9)$.
        \vspace{0.8in}
  \item Find the 95th percentile of the sampling distribution. Interpret in context.
        \vspace{0.8in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\textbf{Context: Battery lifetime.}
Battery lifetimes for a certain brand are right-skewed with $\mu = 480$ hours and
$\sigma = 60$ hours.

\begin{enumerate}
  \item For $n = 4$: can we use the normal model for $\bar{x}$? Explain.
        \writelines{2}
  \item For $n = 36$: check conditions, find $\sigma_{\bar{x}}$, and compute $P(\bar{x} > 490)$.
        \vspace{0.9in}
  \item Find the value $c$ such that $P(\bar{x} < c) = 0.025$ for $n = 36$. Interpret.
        \vspace{0.8in}
  \item For $n = 36$, find $P(|\bar{x} - 480| > 15)$ (probability that the sample mean
        is more than 15 hours away from the true mean).
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}
