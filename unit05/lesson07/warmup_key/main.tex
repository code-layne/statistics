\begin{enumerate}

  \item \ansline{When a random sample of size $n$ is drawn from any population with mean $\mu$ and standard deviation $\sigma$, the sampling distribution of $\bar{x}$ has center $\mu$, spread $\sigma/\sqrt{n}$, and is approximately normal for large $n$ (at least 30) --- even if the population is not normal.}

  \item
        \begin{enumerate}[label=\alph*.]
          \item \ansline{No. $n = 5 < 30$ and the population is right-skewed (not normal). The CLT does not guarantee a normal model.}
          \item \ansline{Yes. $n = 40 \ge 30$; by CLT the sampling distribution is approximately normal.} \ans{$\sigma_{\bar{x}} = 10/\sqrt{40} \approx 1.581$.}
        \end{enumerate}

\end{enumerate}
