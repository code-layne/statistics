\noindent\textbf{Part I --- Multiple Choice} \hfill (2 pts each $\times$ 10 = 20 pts)

\begin{enumerate}

  \item Which of the following is the best description of a \emph{sampling distribution}?
    \begin{enumerate}[label=(\Alph*)]
      \item The distribution of data values in a single sample.
      \item The distribution of all possible values of a statistic over all possible samples.
      \item A histogram of the population from which samples are drawn.
      \item The probability that a sample mean equals the population mean.
      \item The distribution of sample sizes used in a study.
    \end{enumerate}

  \vspace{0.05in}

  \item Heights of adult males are approximately $N(69.2\text{ in}, 3.6\text{ in})$.
        What is $P(X > 75)$?
    \begin{enumerate}[label=(\Alph*)]
      \item 0.9463 \item 0.0537 \item 0.1611 \item 0.8389 \item 0.0268
    \end{enumerate}

  \item A population has $\mu = 50$ and $\sigma = 10$. A random sample of $n = 25$ is
        taken. What is $\sigma_{\bar{x}}$?
    \begin{enumerate}[label=(\Alph*)]
      \item 10 \item 5 \item 2 \item 0.4 \item 50
    \end{enumerate}

  \item The Central Limit Theorem guarantees that the sampling distribution of $\bar{x}$
        is approximately normal when which condition is met?
    \begin{enumerate}[label=(\Alph*)]
      \item The population is normally distributed.
      \item $n \ge 10$.
      \item $n \ge 30$ (or the population is normal).
      \item $np \ge 10$ and $n(1-p) \ge 10$.
      \item The sample size is at least 5\% of the population.
    \end{enumerate}

  \item Which of the following statistics is \emph{not} an unbiased estimator of its
        corresponding population parameter?
    \begin{enumerate}[label=(\Alph*)]
      \item $\bar{x}$ as an estimator of $\mu$.
      \item $\hat{p}$ as an estimator of $p$.
      \item $s^2$ (computed with $n-1$) as an estimator of $\sigma^2$.
      \item The sample maximum as an estimator of the population maximum.
      \item The sample proportion as an estimator of the population proportion.
    \end{enumerate}

  \item In a large city, 40\% of residents have a gym membership ($p = 0.40$). A random
        sample of $n = 100$ residents is selected. The standard deviation of the sampling
        distribution of $\hat{p}$ is:
    \begin{enumerate}[label=(\Alph*)]
      \item 0.40 \item 0.0490 \item 0.24 \item 0.16 \item 0.0024
    \end{enumerate}

  \item A random sample of $n = 64$ adults is selected from a population with $\mu = 120$
        and $\sigma = 16$. Which of the following is the probability $P(\bar{x} > 122)$?
    \begin{enumerate}[label=(\Alph*)]
      \item 0.8413 \item 0.1587 \item 0.4013 \item 0.0548 \item 0.3085
    \end{enumerate}

  \item To use the normal model for the sampling distribution of $\hat{p}$, which condition
        must be verified?
    \begin{enumerate}[label=(\Alph*)]
      \item $n \ge 30$
      \item $\sigma > 0$
      \item $np \ge 10$ and $n(1-p) \ge 10$
      \item The population is symmetric.
      \item $p > 0.5$
    \end{enumerate}

  \item A researcher takes random samples of size $n = 50$ from Group 1 ($\mu_1 = 80$,
        $\sigma_1 = 12$) and $n = 50$ from Group 2 ($\mu_2 = 75$, $\sigma_2 = 10$). The
        mean of the sampling distribution of $\bar{x}_1 - \bar{x}_2$ is:
    \begin{enumerate}[label=(\Alph*)]
      \item 155 \item 5 \item $-5$ \item 7.81 \item 2.24
    \end{enumerate}

  \item Which of the following best explains why a larger sample size produces a more
        precise estimate of a population parameter?
    \begin{enumerate}[label=(\Alph*)]
      \item Larger samples eliminate bias.
      \item Larger samples increase $\sigma_{\bar{x}}$.
      \item Larger samples reduce the standard error, narrowing the sampling distribution.
      \item Larger samples guarantee a normal population.
      \item Larger samples make the statistic equal to the parameter.
    \end{enumerate}

\end{enumerate}

\noindent\textbf{Part II --- Free Response} \hfill (see point values)

\begin{enumerate}[resume]

  \item \textbf{(12 pts)} According to national data, 28\% of adults exercise at least
        5 days per week ($p = 0.28$). A health researcher surveys a random sample of
        $n = 200$ adults from a large population.

        \begin{enumerate}[label=(\alph*)]
          \item Check all three conditions for using the normal model for $\hat{p}$.
                State each condition and show that it is (or is not) satisfied. \hfill (3 pts)
                \vspace{1.3in}
          \item Find $\mu_{\hat{p}}$ and $\sigma_{\hat{p}}$. \hfill (2 pts)
                \vspace{0.8in}
          \item Find $P(\hat{p} > 0.32)$. Show all steps: standardize, and state the
                probability. \hfill (4 pts)
                \vspace{1.2in}
          \item Interpret $P(\hat{p} > 0.32)$ in a complete sentence in context. \hfill (3 pts)
                \writelines{2}
        \end{enumerate}

  \item \textbf{(18 pts)} A manufacturer produces bolts with lengths that are approximately
        normally distributed with $\mu = 50$ mm and $\sigma = 1.5$ mm.

        \begin{enumerate}[label=(\alph*)]
          \item What is the probability that a single randomly selected bolt has length
                greater than 51.5 mm? \hfill (3 pts)
                \vspace{0.9in}
          \item A quality inspector measures a random sample of $n = 36$ bolts each shift.
                Check all three conditions for using the normal model for $\bar{x}$. \hfill (3 pts)
                \vspace{1.0in}
          \item Find $\mu_{\bar{x}}$ and $\sigma_{\bar{x}}$. \hfill (2 pts)
                \vspace{0.6in}
          \item Find $P(\bar{x} > 50.5)$. \hfill (4 pts)
                \vspace{1.0in}
          \item Compare your answers to (a) and (d). Explain why the probability for
                $\bar{x}$ is much smaller than the probability for a single bolt. \hfill (3 pts)
                \writelines{3}
          \item The manufacturer is concerned if the sample mean falls below 49.4 mm.
                Find this probability and determine if this should be considered an unusual
                event. \hfill (3 pts)
                \vspace{1.0in}
        \end{enumerate}

  \item \textbf{(10 pts)} A study compares the proportion of students who complete
        homework in two schools. At School 1, 65\% complete homework ($p_1 = 0.65$,
        $n_1 = 120$). At School 2, 58\% complete homework ($p_2 = 0.58$, $n_2 = 150$).
        The samples are independent random samples from large student populations.

        \begin{enumerate}[label=(\alph*)]
          \item Verify the four Large Counts conditions. \hfill (4 pts)
                \vspace{0.8in}
          \item Find the mean and standard deviation of the sampling distribution of
                $\hat{p}_1 - \hat{p}_2$. \hfill (3 pts)
                \vspace{0.7in}
          \item Find $P(\hat{p}_1 - \hat{p}_2 > 0.15)$. \hfill (3 pts)
                \vspace{0.9in}
        \end{enumerate}

\end{enumerate}
