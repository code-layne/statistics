\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\textbf{Context: Baby weights.} $N(7.5, 1.1)$
\begin{enumerate}
  \item \ans{$\mu \pm 2\sigma = 7.5 \pm 2(1.1) = 7.5 \pm 2.2$; the range is $(5.3, 9.7)$ lb.}
  \item \ans{$z = (6.0 - 7.5)/1.1 \approx -1.36$. This is about 1.4 SD below the mean --- not extremely unusual (within 2 SD), but below average.}
  \item \ansline{There is approximately an 8.6\% chance that a randomly selected newborn weighs less than 6.0 lb. In other words, about 8.6\% of newborns in the U.S. weigh less than 6 pounds.}
\end{enumerate}
\end{tcolorbox}

\vspace{0.1in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\textbf{Context: State exam.} $N(72, 9)$
\begin{enumerate}
  \item \ans{$z = (85-72)/9 \approx 1.44$; $P(X>85) = 1 - 0.9251 = 0.0749$. About 7.5\% of students score above 85.}
  \item \ans{$z_1 = (60-72)/9 \approx -1.33$, $z_2 = (80-72)/9 \approx 0.89$; $P(60<X<80) = P(-1.33<Z<0.89) = 0.8133 - 0.0918 = 0.7215$. About 72\%.}
  \item \ans{$P(X > x) = 0.10 \Rightarrow z = 1.28$; $x = 72 + 1.28(9) \approx 83.5$. The minimum score is approximately 83.5 (or 84).}
  \item \ansline{A student must score at least 83.5 on the state exam to be in the top 10\% and qualify for the scholarship.}
\end{enumerate}
\end{tcolorbox}

\vspace{0.1in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\textbf{Context: Daily rainfall.} $N(1.2, 0.4)$
\begin{enumerate}
  \item \ans{$z_1 = (0.8-1.2)/0.4 = -1.00$, $z_2 = (2.0-1.2)/0.4 = 2.00$; $P(0.8<X<2.0) = 0.9772 - 0.1587 = 0.8185$. About 81.9\%.}
  \item \ans{5th percentile: $z = -1.645$; $x = 1.2 + (-1.645)(0.4) \approx 0.54$ cm. 95th percentile: $z = 1.645$; $x \approx 1.86$ cm. On 5\% of days it rains less than 0.54 cm and on 5\% of days it rains more than 1.86 cm.}
  \item \ans{Most extreme 2\%: split into $1\%$ in each tail. $z = \pm 2.326$; lower: $1.2 - 2.326(0.4) \approx 0.27$ cm; upper: $1.2 + 2.326(0.4) \approx 2.13$ cm. Days with rainfall below 0.27 cm or above 2.13 cm trigger weather alerts.}
  \item \ansline{Rainfall data is often right-skewed (many low-rain days and occasional high-rain days), so the normal model may not be appropriate. To check: examine a histogram or normal probability plot for symmetry and bell shape. If the data are skewed, a different model (e.g., exponential or gamma) might be better.}
\end{enumerate}
\end{tcolorbox}
