\begin{enumerate}

  \item Reading for pleasure, men vs.\ women.
        \begin{enumerate}[label=\alph*.]
          \item \ans{Random: both stated. 10\%: both $\le 10\%$ of national adult population. Independent groups. Large Counts: $200(0.48)=96$, $200(0.52)=104$, $220(0.61)=134.2$, $220(0.39)=85.8$ --- all $\ge 10$. All conditions met.}
          \item \ans{$\mu = 0.48-0.61 = -0.13$; $\sigma = \sqrt{0.48(0.52)/200 + 0.61(0.39)/220} = \sqrt{0.001248+0.001081} \approx \sqrt{0.002329} \approx 0.04826$.}
          \item \ans{$z = (-0.20-(-0.13))/0.04826 \approx -1.451$; $P \approx 0.0735$. About 7.4\%.}
          \item \ansline{There is approximately a 7.4\% chance that, in random samples of 200 men and 220 women, the sample proportion of male readers falls at least 20 percentage points below the sample proportion of female readers --- even if the true difference is only 13 points.}
        \end{enumerate}

  \item Composting.
        \begin{enumerate}[label=\alph*.]
          \item \ans{$160(0.25)=40$; $160(0.75)=120$; $200(0.18)=36$; $200(0.82)=164$. All $\ge 10$.}
          \item \ans{$\mu = 0.25-0.18 = 0.07$; $\sigma = \sqrt{0.25(0.75)/160 + 0.18(0.82)/200} = \sqrt{0.001172+0.000738} \approx \sqrt{0.001910} \approx 0.04370$. $z = (0.12-0.07)/0.04370 \approx 1.144$; $P > 0.12 \approx 1-0.8737 = 0.1263 \approx 12.6\%$.}
        \end{enumerate}

\end{enumerate}

\begin{extensionbox}
\textbf{Extension.}
\ans{$\sigma = \sqrt{0.50(0.50)/n + 0.50(0.50)/n} = \sqrt{0.25/n + 0.25/n} = \sqrt{0.50/n}$. For $\sigma \le 0.05$: $0.50/n \le 0.0025 \Rightarrow n \ge 200$.}

\begin{teachernote}
The problem states ``show $= \sqrt{1/n}$'' but the exact expression is $\sqrt{0.50/n} = \sqrt{1/(2n)}$, not $\sqrt{1/n}$. The minimum is $n = 200$.
\end{teachernote}
\end{extensionbox}
