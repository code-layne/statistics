\begin{enumerate}

  \item In a large city, 45\% of adults prefer coffee over tea ($p = 0.45$). A random
        sample of $n = 160$ adults is selected. Check conditions and find
        $\mu_{\hat{p}}$, $\sigma_{\hat{p}}$, and $P(\hat{p} > 0.50)$.
        \vspace{1.5in}

  \item A researcher wants to compare the proportion of coffee-preferrers in two cities.
        City 1 has $p_1 = 0.45$ and City 2 has $p_2 = 0.38$. If you sampled 100 people
        from each city, what would you expect $\hat{p}_1 - \hat{p}_2$ to equal, on
        average? Explain your reasoning.
        \writelines{2}

\end{enumerate}
