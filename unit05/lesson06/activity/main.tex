\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\textbf{Context: Seatbelt use by age group.}
Among adults 18--34, 82\% always wear a seatbelt ($p_1 = 0.82$, $n_1 = 150$).
Among adults 35--54, 88\% always wear a seatbelt ($p_2 = 0.88$, $n_2 = 150$).

\begin{enumerate}
  \item Check the four Large Counts conditions.
        \vspace{0.7in}
  \item Find $\mu_{\hat{p}_1-\hat{p}_2}$ and $\sigma_{\hat{p}_1-\hat{p}_2}$.
        \vspace{0.6in}
  \item Find $P(\hat{p}_1 - \hat{p}_2 < -0.10)$.
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\textbf{Context: Online vs.\ in-store shopping.}
In a large metro area, 55\% of millennials prefer online shopping ($p_1 = 0.55$,
$n_1 = 200$) and 40\% of Gen X prefer online shopping ($p_2 = 0.40$, $n_2 = 180$).

\begin{enumerate}
  \item Check all conditions. Is the normal model appropriate?
        \writelines{3}
  \item Find $\mu$ and $\sigma$ for the sampling distribution of $\hat{p}_1 - \hat{p}_2$.
        \vspace{0.5in}
  \item Find $P(\hat{p}_1 - \hat{p}_2 > 0.20)$.
        \vspace{0.9in}
  \item Find the probability that the difference in sample proportions is less than 0.
        What would this mean in context?
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\textbf{Context: Clinical trial --- treatment vs.\ placebo.}
In a clinical trial, 30\% of placebo patients improved ($p_2 = 0.30$, $n_2 = 100$)
and 50\% of treatment patients improved ($p_1 = 0.50$, $n_1 = 100$).

\begin{enumerate}
  \item Check all conditions.
        \writelines{2}
  \item Compute $\mu$ and $\sigma$ for $\hat{p}_1 - \hat{p}_2$.
        \vspace{0.5in}
  \item Find $P(\hat{p}_1 - \hat{p}_2 > 0.30)$.
        \vspace{0.8in}
  \item Find $P(\hat{p}_1 - \hat{p}_2 < 0)$ and interpret. What does it mean if the
        observed difference is negative in this study?
        \vspace{0.9in}
  \item If the trial is repeated with $n_1 = n_2 = 400$, how does $\sigma$ change?
        Show your calculation.
        \vspace{0.7in}
\end{enumerate}
\end{tcolorbox}
