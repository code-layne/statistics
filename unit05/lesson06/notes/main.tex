\begin{objectivebox}
\textbf{Learning Targets}
\begin{enumerate}
  \item I can state the formulas for center and spread of the sampling distribution of $\hat{p}_1 - \hat{p}_2$.
  \item I can check all four Large Counts conditions.
  \item I can compute probabilities for $\hat{p}_1 - \hat{p}_2$ using the normal model.
\end{enumerate}
\end{objectivebox}

\begin{notesbox}
\textbf{Conditions for $\hat{p}_1 - \hat{p}_2$}

\begin{enumerate}
  \item \textbf{Random (Group 1):} sample 1 is a \blank{2.0in}.
  \item \textbf{Random (Group 2):} sample 2 is a \blank{2.0in}.
  \item \textbf{10\% (Independence within each group):} $n_1 \le 0.10 N_1$ and $n_2 \le 0.10 N_2$.
  \item \textbf{Independence between groups:} the two samples are \blank{2.0in}.
  \item \textbf{Large Counts (all four):}
        $n_1 p_1 \ge $ \blank{0.3in},\quad $n_1(1-p_1) \ge $ \blank{0.3in},\quad
        $n_2 p_2 \ge $ \blank{0.3in},\quad $n_2(1-p_2) \ge $ \blank{0.3in}
\end{enumerate}
\end{notesbox}
