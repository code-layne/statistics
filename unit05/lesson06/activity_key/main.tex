\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\begin{enumerate}
  \item \ans{$n_1 p_1=123$, $n_1(1-p_1)=27$, $n_2 p_2=132$, $n_2(1-p_2)=18$. All $\ge 10$.}
  \item \ans{$\mu = 0.82 - 0.88 = -0.06$; $\sigma = \sqrt{0.82(0.18)/150 + 0.88(0.12)/150} = \sqrt{0.000984+0.000704} \approx \sqrt{0.001688} \approx 0.04108$.}
  \item \ans{$z = (-0.10-(-0.06))/0.04108 \approx -0.973$; $P < -0.10 \approx 0.1653$. About 16.5\%.}
\end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\begin{enumerate}
  \item \ans{Random: both stated. 10\%: both $\le 10\%$ of large metro population. Independent groups. Large Counts: $110, 90, 72, 108$ --- all $\ge 10$. Normal model appropriate.}
  \item \ans{$\mu = 0.55 - 0.40 = 0.15$; $\sigma = \sqrt{0.55(0.45)/200 + 0.40(0.60)/180} = \sqrt{0.001238 + 0.001333} \approx 0.05070$.}
  \item \ans{$z = (0.20-0.15)/0.05070 \approx 0.986$; $P > 0.20 \approx 1 - 0.8379 = 0.1621 \approx 16.2\%$.}
  \item \ans{$z = (0-0.15)/0.05070 \approx -2.959$; $P < 0 \approx 0.0015$. A negative difference would mean the Gen X sample proportion exceeds the millennial proportion --- very unlikely but possible due to sampling variability.}
\end{enumerate}
\end{tcolorbox}

\vspace{0.1in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\begin{enumerate}
  \item \ans{Random: stated (random assignment). Independence: subjects assigned independently. Large Counts: $50, 50, 30, 70$ --- all $\ge 10$. Conditions met.}
  \item \ans{$\mu = 0.50 - 0.30 = 0.20$; $\sigma = \sqrt{0.50(0.50)/100 + 0.30(0.70)/100} = \sqrt{0.0025+0.0021} = \sqrt{0.0046} \approx 0.06782$.}
  \item \ans{$z = (0.30-0.20)/0.06782 \approx 1.475$; $P > 0.30 \approx 1 - 0.9299 = 0.0701 \approx 7.0\%$.}
  \item \ans{$z = (0-0.20)/0.06782 \approx -2.950$; $P < 0 \approx 0.0016$. A negative difference would mean the placebo group had a higher improvement rate than treatment --- strong evidence (if observed) that the treatment may be harmful, not helpful.}
  \item \ans{$\sigma = \sqrt{0.50(0.50)/400 + 0.30(0.70)/400} = \sqrt{0.000625+0.000525} = \sqrt{0.00115} \approx 0.03391$. Approximately halved (from 0.06782 to 0.03391) when $n$ quadrupled.}
\end{enumerate}
\end{tcolorbox}
