\begin{enumerate}
  \item \ansline{Biased high. The sample maximum consistently overestimates the population maximum; $E(\text{sample max}) > \text{population max}$.}

  \item \ans{$n=16$: $\sigma_{\bar{x}} = 16/\sqrt{16} = 4$. $n=64$: $\sigma_{\bar{x}} = 16/\sqrt{64} = 2$. Quadrupling $n$ cuts $\sigma_{\bar{x}}$ in half.}

  \item \ansline{Any two of: sample mean $\bar{x}$ (estimates $\mu$); sample proportion $\hat{p}$ (estimates $p$); sample variance $s^2$ (estimates $\sigma^2$).}
\end{enumerate}
