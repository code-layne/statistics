\begin{enumerate}

  \item A population has $\mu = 72$ and $\sigma = 18$. A random sample of $n = 81$ is taken.
        \begin{enumerate}[label=\alph*.]
          \item Find $\mu_{\bar{x}}$ and $\sigma_{\bar{x}}$.
                \vspace{0.6in}
          \item Is the sampling distribution of $\bar{x}$ approximately normal? Justify.
                \writelines{2}
          \item Find $P(\bar{x} > 74)$.
                \vspace{0.8in}
        \end{enumerate}

  \item A researcher computes $\bar{x} = 68.4$ from a random sample of $n = 36$ students.
        She uses this as an estimate of the population mean $\mu$. What do we call this
        single computed value (as opposed to the rule used to compute it)?
        \writelines{1}

\end{enumerate}
