\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier R --- Remediate}]
\textbf{Context: Lefties.}
About 10\% of people are left-handed ($p = 0.10$). A random sample of $n = 200$ people
is selected from a large population.

\begin{enumerate}
  \item Check all three conditions for using the normal model for $\hat{p}$.
        \writelines{4}
  \item Find $\mu_{\hat{p}}$ and $\sigma_{\hat{p}}$.
        \vspace{0.6in}
  \item Find $P(\hat{p} < 0.08)$.
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier A --- Approaching Proficiency}]
\textbf{Context: Vaccination rates.}
In a large county, 72\% of adults are vaccinated ($p = 0.72$). A public health researcher
surveys $n = 250$ randomly selected adults.

\begin{enumerate}
  \item Check all three conditions and conclude whether the normal model is appropriate.
        \writelines{3}
  \item Find $\mu_{\hat{p}}$ and $\sigma_{\hat{p}}$.
        \vspace{0.5in}
  \item Find $P(0.68 \le \hat{p} \le 0.76)$.
        \vspace{0.9in}
  \item What sample proportion is at the 90th percentile of the sampling distribution?
        Interpret in context.
        \vspace{0.9in}
\end{enumerate}
\end{tcolorbox}

\vspace{0.12in}

\begin{tcolorbox}[colback=redbg, colframe=black!40, boxrule=0.6pt, arc=2mm,
    left=3mm, right=3mm, top=2mm, bottom=2mm, title=\textbf{Tier E --- Extension}]
\textbf{Context: Defective products.}
A factory has a 5\% defect rate ($p = 0.05$). An auditor plans to inspect a random
sample to estimate $\hat{p}$.

\begin{enumerate}
  \item For $n = 60$: check the Large Counts condition. Is the normal model appropriate?
        \writelines{2}
  \item For $n = 200$: check conditions and compute $P(\hat{p} > 0.08)$.
        \vspace{0.9in}
  \item How large must $n$ be so that $\sigma_{\hat{p}} \le 0.01$?
        \vspace{0.7in}
  \item Describe how the sampling distribution of $\hat{p}$ changes as $n$ increases from
        60 to 200 to 500 in terms of shape, center, and spread.
        \writelines{3}
\end{enumerate}
\end{tcolorbox}
