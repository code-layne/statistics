\begin{enumerate}

  \item A random sample of 64 students finds that 20 own a pet. Calculate $\hat{p}$.
        \vspace{0.5in}

  \item In a population, 35\% of adults exercise daily ($p = 0.35$). A random sample of
        $n = 200$ is selected.
        \begin{enumerate}[label=\alph*.]
          \item Is $\hat{p}$ an unbiased estimator of $p$? What does that mean for the
                center of the sampling distribution of $\hat{p}$?
                \writelines{2}
          \item Would you expect more variability in $\hat{p}$ from samples of $n = 50$
                or $n = 200$? Why?
                \writelines{2}
        \end{enumerate}

\end{enumerate}
