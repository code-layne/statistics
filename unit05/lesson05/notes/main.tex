\begin{objectivebox}
\textbf{Learning Targets}
\begin{enumerate}
  \item I can state $\mu_{\hat{p}} = p$ and $\sigma_{\hat{p}} = \sqrt{p(1-p)/n}$.
  \item I can check the conditions for using the normal model for $\hat{p}$.
  \item I can find probabilities involving $\hat{p}$ using the normal approximation.
\end{enumerate}
\end{objectivebox}

Check all three before computing any probability:
\begin{enumerate}
  \item \textbf{Random:} \blank{2.5in}
  \item \textbf{10\% condition (Independence):} $n \le 0.10 \cdot N$ where $N$ = \blank{1.5in}
  \item \textbf{Large Counts:} $np \ge $ \blank{0.4in} \textbf{and} $n(1-p) \ge$ \blank{0.4in}
