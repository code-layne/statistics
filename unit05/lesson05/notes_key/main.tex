\begin{vocabbox}
\begin{description}
  \item[Sampling distribution of $\hat{p}$] all possible values of \ans{$\hat{p}$} from samples of size \ans{$n$}.
  \item[Large Counts condition] $np \ge$ \ans{10} and $n(1-p) \ge$ \ans{10}.
  \item[10\% condition] $n \le$ \ans{10\%} of the population.
\end{description}
\end{vocabbox}

\begin{notesbox}
\textbf{Sampling Distribution of $\hat{p}$}
\begin{itemize}
  \item Center: \ans{$p$}
  \item Spread: \ans{$\sqrt{p(1-p)/n}$}
  \item Shape: approximately normal when \ans{$np \ge 10$} and \ans{$n(1-p) \ge 10$}
\end{itemize}
\end{notesbox}

\begin{notesbox}
\textbf{Worked Example: Library cards, $p = 0.62$, $n = 150$}
\begin{enumerate}
  \item \ans{Random: stated.}
  \item \ans{10\%: $150 \le 10\%$ of adults in a large city (thousands) --- yes.}
  \item \ans{$np = 150(0.62) = 93 \ge 10$; $n(1-p) = 150(0.38) = 57 \ge 10$. Both met.}
\end{enumerate}
$\mu_{\hat{p}} = $ \ans{0.62}; $\sigma_{\hat{p}} = $ \ans{$\sqrt{0.62(0.38)/150} \approx 0.03965$}

$P(\hat{p} > 0.65)$: \ans{$z = (0.65-0.62)/0.03965 \approx 0.757$; $P(Z>0.757) \approx 1-0.7754 = 0.2246 \approx 22.5\%$.}
\end{notesbox}

\begin{practicebox}
\textbf{Quick Check: Bus riders, $p = 0.40$, $n = 80$}
\begin{enumerate}
  \item \ans{Random: stated. 10\%: $80 \le 10\%$ of large school enrollment. Large Counts: $np = 32 \ge 10$; $n(1-p) = 48 \ge 10$. All conditions met.}
  \item \ans{$\mu_{\hat{p}} = 0.40$; $\sigma_{\hat{p}} = \sqrt{0.40(0.60)/80} = \sqrt{0.003} \approx 0.05477$.}
  \item \ans{$z = (0.50 - 0.40)/0.05477 \approx 1.826$; $P(\hat{p} \ge 0.50) \approx 1 - 0.9660 = 0.0340$. About 3.4\%.}
\end{enumerate}
\end{practicebox}
