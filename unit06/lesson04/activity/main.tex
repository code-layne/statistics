\textit{For each scenario, (a) state $H_0$ and $H_a$, (b) identify the test as one-sided or
two-sided and justify, (c) verify conditions, and (d) calculate the test statistic $z$.}

\begin{scenariobox}[Scenario 1 --- Voter Turnout]{sky}
A political analyst claims that voter turnout in a city is 40\%. A civic group believes
turnout has \emph{changed} from 40\%. They survey 300 randomly selected registered voters
and find 105 voted in the last election.

\begin{enumerate}[label=\alph*.]
  \item $H_0$: \blank{4cm} \quad $H_a$: \blank{4cm} \quad One-sided or two-sided? \blank{3cm}
  \item Conditions:
        \begin{itemize}
          \item Random: \blank{5cm}
          \item Large Counts: \blank{7cm}
          \item 10\%: \blank{5cm}
        \end{itemize}
  \item $\hat p = $ \blank{2cm} \quad $z = \dfrac{\blank{1.5cm} - \blank{1.5cm}}{\sqrt{\blank{2.5cm}/\blank{1.5cm}}} = $ \blank{2cm}
\end{enumerate}
\end{scenariobox}

\begin{scenariobox}[Scenario 2 --- Medication Effectiveness]{sky}
A pharmaceutical company states that 75\% of patients improve with their medication. An
independent researcher thinks the true rate is \emph{lower}. She randomly selects 160
patients from clinical records and finds 108 improved.

\begin{enumerate}[label=\alph*.]
  \item $H_0$: \blank{4cm} \quad $H_a$: \blank{4cm} \quad One-sided or two-sided? \blank{3cm}
  \item Conditions (check all three): \writelines{3}
  \item $\hat p = $ \blank{2cm} \quad $z = $ \blank{4cm} $\approx$ \blank{2cm}
\end{enumerate}
\end{scenariobox}

\begin{scenariobox}[Scenario 3 --- Organic Milk]{sky}
Industry data suggest 30\% of households buy organic milk. A grocery chain manager believes
more than 30\% of \emph{her} store's shoppers buy organic milk. She randomly selects 90
shoppers and finds 34 buy organic milk.

\begin{enumerate}[label=\alph*.]
  \item State hypotheses and justify the direction of $H_a$.
        \writelines{2}
  \item Verify conditions.
        \writelines{3}
  \item Compute $z$.
        \writelines{2}
\end{enumerate}
\end{scenariobox}

\begin{reflectionbox}
Scenarios 1--3 cover all three possible directions for $H_a$. In your own words, explain the
rule for choosing a one-sided vs.\ two-sided alternative. When might a researcher be tempted
to choose the direction \emph{after} seeing the data, and why is that a problem?
\writelines{3}
\end{reflectionbox}
