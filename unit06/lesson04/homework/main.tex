\textit{For each problem, state $H_0$ and $H_a$, verify conditions, and compute $z$.}

\begin{enumerate}

  \item The CDC reports 18\% of U.S.\ adults smoke cigarettes. A public health researcher
        wants to test whether the smoking rate in a particular county differs from the national
        rate. She randomly surveys 450 county adults and finds 72 are smokers.
        \begin{enumerate}[label=\alph*.]
          \item $H_0$: \blank{5cm} \quad $H_a$: \blank{5cm} \quad Type: \blank{2cm}
          \item Conditions: \writelines{3}
          \item $\hat p = $ \blank{2cm} \quad $z = $ \blank{6cm} $\approx$ \blank{2cm}
        \end{enumerate}

  \item A seed company claims that 85\% of its seeds germinate. An agricultural researcher
        believes the true germination rate is \emph{lower} than claimed. She plants 200
        randomly selected seeds and 162 germinate.
        \begin{enumerate}[label=\alph*.]
          \item State hypotheses and justify the direction of $H_a$. \writelines{2}
          \item Verify conditions. \writelines{3}
          \item $\hat p = $ \blank{2cm} \quad $z \approx$ \blank{2cm}
        \end{enumerate}

  \item A school claims that 70\% of students participate in extracurricular activities.
        A parent group believes participation has \emph{increased}. A random sample of
        180 students finds 135 participate.
        \begin{enumerate}[label=\alph*.]
          \item Hypotheses: \writelines{1}
          \item Conditions: \writelines{2}
          \item $z \approx$ \blank{2cm}
        \end{enumerate}

  \item \textbf{Reflection:} In Problem 1, what changes in the setup if the researcher
        instead believed the county rate was \emph{higher} than 18\%? \writelines{2}

\end{enumerate}

\begin{reflectionbox}
\textbf{Preview:} In Lesson 6.5 you will use the test statistic $z$ to find the $p$-value
and interpret it as evidence against $H_0$. The $p$-value answers: ``If $H_0$ were true,
how likely is a sample result at least this extreme?''
\end{reflectionbox}
