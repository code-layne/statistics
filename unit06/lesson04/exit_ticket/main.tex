A city transportation department reports that 55\% of commuters use public transit. A policy
researcher believes the actual proportion has \emph{fallen below} 55\%. She randomly surveys
240 commuters and finds 120 use public transit.

\begin{enumerate}
  \item State hypotheses.
        $H_0$: \blank{5cm} \quad $H_a$: \blank{5cm}

  \item Verify conditions.
        \begin{itemize}
          \item Random: \blank{6cm}
          \item Large Counts (use $p_0$): \blank{7cm}
          \item 10\% rule: \blank{5cm}
        \end{itemize}

  \item Compute the test statistic. \quad ($\hat p = $ \blank{2cm})

        $z = \dfrac{\hat p - p_0}{\sqrt{p_0(1-p_0)/n}} = \dfrac{\blank{1.5cm} - \blank{1.5cm}}{\sqrt{\blank{2.5cm}/\blank{1.5cm}}} \approx$ \blank{2cm}
\end{enumerate}
