\textbf{Directions:} For each scenario, a confidence interval has already been computed.
Interpret the interval and use it to evaluate the given claim.

\begin{scenariobox}[Scenario 1 --- Teen Driving]{sky}
A 95\% CI for $p_{\text{senior}} - p_{\text{junior}}$, the difference in the proportions of
seniors and juniors who have their own car, is $(0.12, 0.28)$.

\begin{enumerate}[label=\arabic*.]
  \item Write a correct interpretation of this interval. \writelines{2}
  \item Does the interval contain 0? \blank{1.5cm}
  \item Does the interval provide convincing evidence that a higher proportion of seniors
        than juniors have their own car? Justify your answer. \writelines{2}
\end{enumerate}
\end{scenariobox}

\begin{scenariobox}[Scenario 2 --- Homework Completion]{sky}
A school administrator compares the proportion of students who complete homework on time
in two classes. The 90\% CI for $p_A - p_B$ is $(-0.08, 0.14)$.

\begin{enumerate}[label=\arabic*.]
  \item Write a correct interpretation of this interval (make up reasonable context). \writelines{2}
  \item Does the interval provide evidence that the classes differ in homework completion rates?
        Explain using the zero rule. \writelines{2}
\end{enumerate}
\end{scenariobox}

\begin{scenariobox}[Scenario 3 --- Build and Interpret]{sky}
A researcher surveyed random samples to compare the proportions of adults in two cities
who commute by public transportation. City P: $n=320$, $\hat p_P=0.38$. City Q: $n=280$, $\hat p_Q=0.29$.

\begin{enumerate}[label=\arabic*.]
  \item Verify the Large Counts condition. \writelines{2}
  \item Construct a 95\% CI for $p_P - p_Q$. \writelines{3}
  \item Interpret and evaluate: does the interval provide evidence of a difference? \writelines{2}
\end{enumerate}
\end{scenariobox}
