\begin{notesbox}{Interpreting a Two-Sample CI}
\textbf{Template:}\\
``We are \blank{2cm}\% confident that the true difference (\blank{4cm}) in the proportions
of \blank{6cm} and \blank{6cm} who [context] is between \blank{2cm} and \blank{2cm}.''\\[6pt]
\textbf{Example.} 95\% CI for $p_F - p_M$ is $(-0.049, 0.149)$:\\
\writelines{2}
\end{notesbox}

\begin{notesbox}{The Zero Rule}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{@{} p{0.44\linewidth} p{0.50\linewidth} @{}}
\textbf{0 is NOT in the interval} & \textbf{0 IS in the interval} \\
$\Rightarrow$ 0 is \blank{4cm} & $\Rightarrow$ 0 is \blank{4cm} \\
$\Rightarrow$ \blank{5cm} evidence & $\Rightarrow$ \blank{4cm} evidence \\
\quad of a difference & \quad of a difference \\
\end{tabular}\\[6pt]
\textbf{Sign:} If the entire interval is positive $\Rightarrow p_1$ \blank{1cm} $p_2$. \quad
Entirely negative $\Rightarrow p_1$ \blank{1cm} $p_2$.
\end{notesbox}
